\chapter{Integrales dobles y sus aplicaciones}


La integral definida de una variable permite calcular ÃĄreas bajo curvas y acumulaciones a lo largo de intervalos. La integral doble extiende esta idea a funciones de dos variables, permitiendo calcular volÃšmenes bajo superficies, ÃĄreas de regiones planas, masas, centros de masa y otras magnitudes sobre regiones del plano.

\section{Integral doble sobre un rectÃĄngulo}

Sea $R=[a,b]\times[c,d]$ un rectÃĄngulo en el plano. Una particiÃģn de $R$ consiste en dividir $[a,b]$ en $m$ subintervalos de longitudes $\Delta x_i$ y $[c,d]$ en $n$ subintervalos de longitudes $\Delta y_j$, formando $mn$ subrectÃĄngulos $R_{ij}$ de ÃĄrea $\Delta A_{ij}=\Delta x_i\Delta y_j$. Elegimos un punto muestra $(\bar{x}_{ij},\bar{y}_{ij})$ en cada subrectÃĄngulo y formamos la suma de Riemann
\[
\sum_{i=1}^m\sum_{j=1}^n f(\bar{x}_{ij},\bar{y}_{ij})\,\Delta A_{ij}.
\]
Si el lÃ­mite de estas sumas existe cuando la norma de la particiÃģn tiende a cero, se dice que $f$ es integrable sobre $R$ y se define la integral doble
\[
\iint_R f(x,y)\,dA = \lim_{\|P\|\to0}\sum_{i,j} f(\bar{x}_{ij},\bar{y}_{ij})\,\Delta A_{ij}.
\]

\begin{theorem}[Integrabilidad]
Si $f$ es continua en $R$, entonces $f$ es integrable en $R$. MÃĄs generalmente, si $f$ es acotada y continua excepto posiblemente en un nÃšmero finito de curvas suaves, tambiÃĐn es integrable.
\end{theorem}

El cÃĄlculo directo de la integral doble a partir de sumas de Riemann es engorroso. El teorema de Fubini la reduce a dos integrales simples sucesivas.

\begin{theorem}[Fubini]
Si $f$ es continua en el rectÃĄngulo $R=[a,b]\times[c,d]$, entonces
\[
\iint_R f(x,y)\,dA = \int_a^b\left[\int_c^d f(x,y)\,dy\right]dx
= \int_c^d\left[\int_a^b f(x,y)\,dx\right]dy.
\]
\end{theorem}

La primera integral iterada indica que primero se integra respecto a $y$ (manteniendo $x$ fija) y luego respecto a $x$; la segunda invierte el orden.

\section{Propiedades de la integral doble}

\begin{itemize}
    \item \textbf{Linealidad:} $\iint_R [\alpha f+\beta g]\,dA = \alpha\iint_R f\,dA + \beta\iint_R g\,dA$.
    \item \textbf{Aditividad:} si $R=R_1\cup R_2$ y $R_1,R_2$ se solapan solo en sus fronteras, entonces $\iint_R f = \iint_{R_1} f + \iint_{R_2} f$.
    \item \textbf{ComparaciÃģn:} si $f(x,y)\le g(x,y)$ en $R$, entonces $\iint_R f\le \iint_R g$.
\end{itemize}

\begin{pasos}[EvaluaciÃģn de una integral doble sobre un rectÃĄngulo]
\begin{enumerate}
    \item \textbf{Elegir orden de integraciÃģn.} Decidir si se integra primero con respecto a $x$ o a $y$. La elecciÃģn depende de la funciÃģn; en general se busca que la integral interior sea sencilla.
    \item \textbf{Integrar interiormente.} Mantener la otra variable fija y aplicar las tÃĐcnicas de integraciÃģn de una variable.
    \item \textbf{Integrar exteriormente.} Integrar la funciÃģn resultante respecto a la variable restante en el intervalo correspondiente.
\end{enumerate}
\end{pasos}

\begin{example}[Integral doble sobre un rectÃĄngulo]
Calcular $\displaystyle\iint_R (x^2 y + y^2)\,dA$, donde $R=[0,1]\times[0,2]$.
\begin{myproof}
\textbf{Paso 1. Integral iterada.} Usamos el orden $dy\,dx$:
\[
\iint_R (x^2 y + y^2)\,dA
= \int_0^1\int_0^2 (x^2 y + y^2)\,dy\,dx.
\]
\textbf{Paso 2. Integral interior.}
\[
\int_0^2 (x^2 y + y^2)\,dy
= \left[\frac{x^2 y^2}{2} + \frac{y^3}{3}\right]_0^2
= \frac{x^2\cdot 4}{2} + \frac{8}{3}
= 2x^2+\frac{8}{3}.
\]
\textbf{Paso 3. Integral exterior.}
\[
\int_0^1 \left(2x^2+\frac{8}{3}\right)dx
= \left[\frac{2x^3}{3}+\frac{8x}{3}\right]_0^1
= \frac{2}{3}+\frac{8}{3}=\frac{10}{3}.
\]
\textbf{Paso 4. Resultado.}
\[
\boxed{\iint_R (x^2 y + y^2)\,dA = \tfrac{10}{3}.}
\]
\end{myproof}
\end{example}

\section{Integrales dobles sobre regiones generales}

No toda regiÃģn de integraciÃģn es un rectÃĄngulo. Consideramos regiones acotadas con fronteras suaves. Dos tipos bÃĄsicos son:

\begin{definition}[RegiÃģn tipo I ($y$-simple)]
Una regiÃģn $D$ es de tipo I si puede expresarse como
\[
D=\{(x,y): a\le x\le b,\; g_1(x)\le y\le g_2(x)\},
\]
donde $g_1,g_2$ son funciones continuas en $[a,b]$.
\end{definition}

\begin{definition}[RegiÃģn tipo II ($x$-simple)]
Una regiÃģn $D$ es de tipo II si puede expresarse como
\[
D=\{(x,y): c\le y\le d,\; h_1(y)\le x\le h_2(y)\},
\]
donde $h_1,h_2$ son funciones continuas en $[c,d]$.
\end{definition}

Para regiones tipo I, la integral doble se calcula como
\[
\iint_D f(x,y)\,dA = \int_a^b \int_{g_1(x)}^{g_2(x)} f(x,y)\,dy\,dx.
\]
Para regiones tipo II, se tiene
\[
\iint_D f(x,y)\,dA = \int_c^d \int_{h_1(y)}^{h_2(y)} f(x,y)\,dx\,dy.
\]

\begin{pasos}[Integral doble sobre una regiÃģn general]
\begin{enumerate}
    \item \textbf{Identificar el tipo de regiÃģn.} Determinar si $D$ es tipo I (y entre dos curvas) o tipo II (x entre dos curvas). Si es posible, elegir el orden que simplifique los lÃ­mites.
    \item \textbf{Establecer los lÃ­mites.} Escribir los lÃ­mites de integraciÃģn segÃšn el tipo elegido.
    \item \textbf{Evaluar la integral iterada.} Integrar primero respecto a la variable interior y luego respecto a la exterior.
\end{enumerate}
\end{pasos}

\begin{example}[RegiÃģn tipo I]
EvalÃše $\displaystyle\iint_D xy\,dA$, donde $D$ es la regiÃģn acotada por $y=x^2$ y $y=x$, con $0\le x\le 1$.
\begin{myproof}
\textbf{Paso 1. RegiÃģn e integral iterada.} La regiÃģn es tipo I: para cada $x\in[0,1]$, $y$ varÃ­a entre $x^2$ y $x$. Por tanto,
\[
\iint_D xy\,dA = \int_0^1\int_{x^2}^{x} xy\,dy\,dx.
\]
\textbf{Paso 2. Integral interior.}
\[
\int_{x^2}^{x} xy\,dy = x\left[\frac{y^2}{2}\right]_{x^2}^{x}
= \frac{x}{2}(x^2-x^4) = \frac{x^3}{2}-\frac{x^5}{2}.
\]
\textbf{Paso 3. Integral exterior.}
\[
\int_0^1 \left(\frac{x^3}{2}-\frac{x^5}{2}\right)dx
= \frac{1}{2}\left[\frac{x^4}{4}-\frac{x^6}{6}\right]_0^1
= \frac{1}{2}\left(\frac{1}{4}-\frac{1}{6}\right)
= \frac{1}{2}\cdot\frac{1}{12}=\frac{1}{24}.
\]
\textbf{Paso 4. Resultado.}
\[
\boxed{\iint_D xy\,dA = \tfrac{1}{24}.}
\]
\end{myproof}
\end{example}

\begin{example}[RegiÃģn tipo II]
EvalÃše $\displaystyle\iint_D (x+y)\,dA$, donde $D$ es la regiÃģn acotada por $x=y^2$ y $x=y+2$.
\begin{myproof}
\textbf{Paso 1. Identificar la regiÃģn.} Los puntos de intersecciÃģn de $y^2$ y $y+2$ satisfacen $y^2=y+2$, es decir, $y^2-y-2=0$, cuyas soluciones son $y=-1$ y $y=2$. Para $y\in[-1,2]$, la curva de la izquierda es $x=y^2$ y la de la derecha es $x=y+2$ (ya que $y+2\ge y^2$ en ese intervalo). Luego $D$ es tipo II:
\[
D=\{(x,y): -1\le y\le 2,\; y^2\le x\le y+2\}.
\]
\textbf{Paso 2. Integral iterada.}
\[
\iint_D (x+y)\,dA = \int_{-1}^{2}\int_{y^2}^{y+2} (x+y)\,dx\,dy.
\]
\textbf{Paso 3. Integral interior.}
\[
\int_{y^2}^{y+2} (x+y)\,dx
= \left[\frac{x^2}{2}+yx\right]_{y^2}^{y+2}.
\]
Evaluando en $y+2$:
\[
\frac{(y+2)^2}{2}+y(y+2) = \frac{y^2+4y+4}{2}+y^2+2y
= \frac{3y^2}{2}+4y+2.
\]
Evaluando en $y^2$:
\[
\frac{y^4}{2}+y^3.
\]
La diferencia es:
\[
\left(\frac{3y^2}{2}+4y+2\right) - \left(\frac{y^4}{2}+y^3\right)
= -\frac{y^4}{2} - y^3 + \frac{3y^2}{2}+4y+2.
\]
\textbf{Paso 4. Integral exterior.}
\[
\int_{-1}^{2}\left(-\frac{y^4}{2} - y^3 + \frac{3y^2}{2}+4y+2\right)dy
= \left[-\frac{y^5}{10} - \frac{y^4}{4} + \frac{y^3}{2} + 2y^2 + 2y\right]_{-1}^{2}.
\]
Evaluamos en $2$:
\[
-\frac{32}{10} - \frac{16}{4} + \frac{8}{2} + 8 + 4
= -\frac{16}{5} - 4 + 4 + 8 + 4 = -\frac{16}{5} + 12 = \frac{44}{5}.
\]
Evaluamos en $-1$:
\[
-\frac{(-1)^5}{10} - \frac{(-1)^4}{4} + \frac{(-1)^3}{2} + 2(-1)^2 + 2(-1)
= \frac{1}{10} - \frac{1}{4} - \frac{1}{2} + 2 - 2.
\]
Simplificando: $\frac{1}{10} - \frac{1}{4} - \frac{1}{2} = \frac{2}{20} - \frac{5}{20} - \frac{10}{20} = -\frac{13}{20}$.
La diferencia es $\frac{44}{5} - \left(-\frac{13}{20}\right) = \frac{176}{20}+\frac{13}{20} = \frac{189}{20}$.
$\displaystyle\boxed{\iint_D (x+y)\,dA = \tfrac{189}{20}}$.
\end{myproof}
\end{example}

\begin{rem}
En regiones generales, elegir el orden de integraciÃģn adecuado puede simplificar drÃĄsticamente los lÃ­mites. A menudo, una regiÃģn que no es tipo I puede ser tipo II, y viceversa. Si la regiÃģn no es de ninguno de estos tipos, se puede descomponer en varias subregiones.
\end{rem}

\begin{prob}[Integrales dobles sobre rectÃĄngulos]
Calcule las siguientes integrales dobles:
\begin{enumerate}
    \item $\displaystyle\iint_R (x^2+2y)\,dA$, $R=[0,2]\times[0,3]$.
    \item $\displaystyle\iint_R \cos(x+y)\,dA$, $R=[0,\pi/2]\times[0,\pi/2]$.
\end{enumerate}
\begin{myproof}
\textbf{(1)} Integramos en el orden $dy\,dx$:

\textbf{Paso 1. Integral interior en $y$.}
\[
\int_0^3 (x^2+2y)\,dy = \left[x^2 y + y^2\right]_0^3 = 3x^2 + 9.
\]

\textbf{Paso 2. Integral exterior en $x$.}
\[
\int_0^2 (3x^2+9)\,dx = \left[x^3+9x\right]_0^2 = 8+18 = 26.
\]
\textbf{Paso 3. Resultado.}
\[
\boxed{\iint_R (x^2+2y)\,dA = 26.}
\]

\textbf{(2)} Integramos en el orden $dy\,dx$:

\textbf{Paso 1. Integral interior en $y$.}
\[
\int_0^{\pi/2} \cos(x+y)\,dy = \bigl[\sin(x+y)\bigr]_0^{\pi/2} = \sin\!\left(x+\tfrac{\pi}{2}\right)-\sin x = \cos x - \sin x.
\]

\textbf{Paso 2. Integral exterior en $x$.}
\[
\int_0^{\pi/2}(\cos x - \sin x)\,dx = \bigl[\sin x + \cos x\bigr]_0^{\pi/2}
= (1+0)-(0+1) = 0.
\]
\textbf{Paso 3. Resultado.}
\[
\boxed{\iint_R \cos(x+y)\,dA = 0.}
\]
\end{myproof}
\end{prob}

\begin{prob}[Regiones generales]
EvalÃše:
\begin{enumerate}
    \item $\displaystyle\iint_D (x+y)\,dA$, donde $D$ es la regiÃģn acotada por $y=x$ y $y=x^2$.
    \item $\displaystyle\iint_D x\,dA$, donde $D$ es la regiÃģn acotada por $y=x^2$ y $x=y^2$.
    \item $\displaystyle\iint_D xy\,dA$, donde $D$ es la regiÃģn acotada por $y=x$ y $x=4-y^2$ (sugerencia: determine los puntos de intersecciÃģn).
\end{enumerate}
\begin{myproof}
\textbf{(1)} Las curvas se intersectan en $x=x^2$, es decir, $x=0$ y $x=1$. Para $x\in[0,1]$ la regiÃģn es tipo I con $x^2\le y\le x$.

\textbf{Paso 1.}
\[
\int_{x^2}^{x}(x+y)\,dy = \left[xy+\frac{y^2}{2}\right]_{x^2}^{x}
= \left(x^2+\frac{x^2}{2}\right)-\left(x^3+\frac{x^4}{2}\right)
= \frac{3x^2}{2}-x^3-\frac{x^4}{2}.
\]

\textbf{Paso 2.}
\[
\int_0^1\!\left(\frac{3x^2}{2}-x^3-\frac{x^4}{2}\right)dx
= \left[\frac{x^3}{2}-\frac{x^4}{4}-\frac{x^5}{10}\right]_0^1
= \frac{1}{2}-\frac{1}{4}-\frac{1}{10}
= \frac{10-5-2}{20} = \frac{3}{20}.
\]
\textbf{Paso 3. Resultado.}
\[
\boxed{\iint_D (x+y)\,dA = \frac{3}{20}.}
\]

\textbf{(2)} Las curvas $y=x^2$ y $x=y^2$ (es decir, $y=\sqrt{x}$) se intersectan en $x=0$ y $x=1$. Para $x\in[0,1]$: $x^2\le y\le\sqrt{x}$.

\textbf{Paso 1.}
\[
\int_{x^2}^{\sqrt{x}} x\,dy = x(\sqrt{x}-x^2) = x^{3/2}-x^3.
\]

\textbf{Paso 2.}
\[
\int_0^1(x^{3/2}-x^3)\,dx
= \left[\frac{2x^{5/2}}{5}-\frac{x^4}{4}\right]_0^1
= \frac{2}{5}-\frac{1}{4} = \frac{8-5}{20} = \frac{3}{20}.
\]
\textbf{Paso 3. Resultado.}
\[
\boxed{\iint_D x\,dA = \frac{3}{20}.}
\]

\textbf{(3)} Las curvas $y=x$ y $x=4-y^2$ se intersectan cuando $y=4-y^2$, es decir, $y^2+y-4=0$, con raÃ­ces
\[
y_{1,2}=\frac{-1\pm\sqrt{17}}{2}.
\]
Para $y\in[y_1,y_2]$ la regiÃģn es tipo II con $y\le x\le 4-y^2$.

\textbf{Paso 1. Integral interior en $x$.}
\[
\int_{y}^{4-y^2} xy\,dx = \frac{y}{2}\bigl[(4-y^2)^2-y^2\bigr]
= \frac{y}{2}(y^4-9y^2+16).
\]

\textbf{Paso 2. Integral exterior en $y$.}
\[
\iint_D xy\,dA = \frac{1}{2}\int_{y_1}^{y_2}(y^5-9y^3+16y)\,dy
= \frac{1}{2}\left[\frac{y^6}{6}-\frac{9y^4}{4}+8y^2\right]_{y_1}^{y_2}.
\]
Usando la relaciÃģn $y_i^2=4-y_i$ (de $y_i^2+y_i-4=0$) se obtiene que $F(y_i)=\dfrac{76+17y_i}{12}$, de modo que
\[
F(y_2)-F(y_1) = \frac{17(y_2-y_1)}{12} = \frac{17\sqrt{17}}{12}.
\]
\textbf{Paso 3. Resultado.}
\[
\boxed{\iint_D xy\,dA = \frac{17\sqrt{17}}{24}.}
\]
\end{myproof}
\end{prob}

\begin{prob}[Cambio de orden de integraciÃģn]
Dibuje la regiÃģn de integraciÃģn y reescriba la integral invirtiendo el orden:
\[
\int_0^1\int_{x^2}^{x} f(x,y)\,dy\,dx.
\]
\begin{myproof}
\textbf{Paso 1. Describir la regiÃģn.} La regiÃģn original es
\[
D = \{(x,y): 0\le x\le 1,\; x^2\le y\le x\}.
\]
EstÃĄ acotada por la parÃĄbola $y=x^2$ (abajo) y la recta $y=x$ (arriba), con $x\in[0,1]$.

\textbf{Paso 2. Redescribir en tÃĐrminos de $y$.} Para $y\in[0,1]$ fijo, $x$ varÃ­a entre $x=y$ (de la recta) y $x=\sqrt{y}$ (de la parÃĄbola, despejando $x$):
\[
D = \{(x,y): 0\le y\le 1,\; y\le x\le \sqrt{y}\}.
\]

\section{Masa y centro de masa de una lÃĄmina plana}

Consideremos una lÃĄmina delgada que ocupa la regiÃģn \(D\) del plano, con densidad superficial de masa \(\rho(x,y)\) (masa por unidad de ÃĄrea). Si la densidad es variable, la masa total se obtiene sumando (integrando) la masa de cada elemento infinitesimal de ÃĄrea \(dA\):
\[

\]

Para localizar el centro de masa, definimos los momentos estÃĄticos respecto a los ejes:
\[
M_x = \iint_D y\,\rho(x,y)\,dA,\qquad
M_y = \iint_D x\,\rho(x,y)\,dA.
\]
El centro de masa \((\bar{x},\bar{y})\) viene dado por
\[
\bar{x} = \frac{M_y}{m}, \qquad \bar{y} = \frac{M_x}{m}.
\]

\begin{rem}
El momento \(M_y\) mide la tendencia de la lÃĄmina a girar alrededor del eje \(y\); por eso multiplicamos la densidad por la distancia \(x\) al eje. AnÃĄlogamente, \(M_x\) mide el giro alrededor del eje \(x\). El centro de masa es el punto donde se concentrarÃ­a toda la masa para producir los mismos momentos.
\end{rem}

\begin{prob}
Determine el centro de masa de la lÃĄmina de densidad $\delta(x,y)=y+1$ sobre el rectÃĄngulo $[0,4]\times[0,3]$.
\begin{myproof}
\textbf{Paso 1. Masa.}
\[

\]
\textbf{Paso 2. Momento $M_x$.}
\begin{align*}
M_{x}&=\int_{0}^{4}\int_{0}^{3}y(y+1)\,dy\,dx
=\int_{0}^{4}\left[\frac{y^3}{3}+\frac{y^2}{2}\right]_0^3 dx
=\int_{0}^{4}\frac{27}{2}\,dx = 54.
\end{align*}
\textbf{Paso 3. Momento $M_y$.}
\begin{align*}

=\int_{0}^{4}x\cdot\frac{15}{2}\,dx
=\frac{15}{2}\left[\frac{x^2}{2}\right]_0^4 = 60.
\end{align*}
\textbf{Paso 4. Centro de masa.}
\[

\end{myproof}
\end{prob}

\section{Momentos de inercia}

El momento de inercia de una lÃĄmina respecto a un eje mide su resistencia a cambiar su velocidad angular alrededor de ese eje. Para una lÃĄmina con densidad \(\rho(x,y)\):

Una regiÃģn \(D\) se dice \emph{\(r\)-simple} si puede expresarse como

\begin{pasos}[EvaluaciÃģn de una integral doble en coordenadas polares]
\begin{enumerate}

\begin{prob}[AplicaciÃģn combinada]

\[

\textbf{Paso 2. SimetrÃ­a.} La regiÃģn y la densidad son simÃĐtricas respecto al eje $x$, luego $\bar{y}=0$.

=\binom{4}{1}\int_0^{2\pi}\cos^2\theta\,d\theta+\binom{4}{3}\int_0^{2\pi}\cos^4\theta\,d\theta

\]
\[

\]

\[
\bar{x}=\frac{M_y}{m}=\frac{7\pi/4}{5\pi/3}=\frac{7}{4}\cdot\frac{3}{5}=\frac{21}{20}.
\]
\[
\boxed{m=\frac{5\pi}{3},\qquad (\bar{x},\bar{y})=\left(\frac{21}{20},\,0\right).}

\end{myproof}

\section{Cambio de variables en integrales dobles}
\label{sec:cambio-var-dobles}
% ============================================================

En muchas integrales dobles la regiÃģn o el integrando sugieren un cambio
de variables que simplifica el cÃĄlculo. El instrumento que mide cÃģmo se
transforma el elemento de ÃĄrea es el \textbf{jacobiano}.

\begin{definition}[TransformaciÃģn y jacobiano en $\mathbb{R}^2$]
Sea $T\colon R^*\subset\mathbb{R}^2\to\mathbb{R}^2$ la transformaciÃģn
$T(u,v)=(x(u,v),\,y(u,v))$ de clase $\mathcal{C}^1$. El
\textbf{jacobiano} de $T$ en $(u,v)$ es el determinante
\[
\frac{\partial(x,y)}{\partial(u,v)}
= \det\begin{pmatrix}
    \dfrac{\partial x}{\partial u} & \dfrac{\partial x}{\partial v} \\[6pt]
    \dfrac{\partial y}{\partial u} & \dfrac{\partial y}{\partial v}
  \end{pmatrix}
= \frac{\partial x}{\partial u}\,\frac{\partial y}{\partial v}
  - \frac{\partial x}{\partial v}\,\frac{\partial y}{\partial u}.
\]
GeomÃĐtricamente, $\left|\dfrac{\partial(x,y)}{\partial(u,v)}\right|$ es
el factor de escala local de ÃĄreas bajo $T$.
\end{definition}

\begin{theorem}[Cambio de variables en integrales dobles]
Sea $T(u,v)=(x(u,v),y(u,v))$ una transformaciÃģn $\mathcal{C}^1$
inyectiva (excepto posiblemente en la frontera) que lleva $R^*$ en
el plano $(u,v)$ a $R$ en el plano $(x,y)$. Si $f$ es continua en
$R$, entonces
\[
\iint_R f(x,y)\,dA
= \iint_{R^*} f\bigl(x(u,v),y(u,v)\bigr)
  \left|\frac{\partial(x,y)}{\partial(u,v)}\right|\,du\,dv.
\]
\end{theorem}

\begin{rem}
El jacobiano puede anularse en un conjunto de ÃĄrea nula (p.ej., el
origen en coordenadas polares) sin invalidar el teorema.
\end{rem}

\begin{example}[Coordenadas polares como caso del Jacobiano]
Calcular $\displaystyle\iint_R e^{x^2+y^2}\,dA$ donde $R$ es el disco
$x^2+y^2\le 4$.
\begin{myproof}
\textbf{Paso 1. TransformaciÃģn.}
$x=r\cos\theta$, $\;y=r\sen\theta$.

\textbf{Paso 2. Jacobiano.}
\[
\frac{\partial(x,y)}{\partial(r,\theta)}
= \det\begin{pmatrix}
    \cos\theta & -r\sen\theta \\
    \sen\theta &  r\cos\theta
  \end{pmatrix}
= r\cos^2\theta + r\sen^2\theta = r.
\]

\textbf{Paso 3. Nueva regiÃģn.}
$x^2+y^2\le4$ se convierte en $0\le r\le2$, $\;0\le\theta\le2\pi$.

\textbf{Paso 4. Integral.}
\[
\iint_R e^{x^2+y^2}\,dA
= \int_0^{2\pi}\!\int_0^2 e^{r^2}\,r\,dr\,d\theta
= 2\pi\Bigl[\tfrac{1}{2}e^{r^2}\Bigr]_0^2
= \pi(e^4-1).
\]
$\boxed{\displaystyle\iint_R e^{x^2+y^2}\,dA = \pi(e^4-1)}$
\end{myproof}
\end{example}
