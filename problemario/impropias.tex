\chapter{Integrales impropias}

\begin{prob}[Trompeta de Gabriel] Dada la región acotada por la gráfica de $f(x) = 1/x$, el eje $x$ y la recta $x=1$, para $x \geq 1$.
\begin{enumerate}
  \item Demuestre que el volumen del sólido generado al girar esta región alrededor del eje $x$ es finito e igual a $\pi$.
  \item Demuestre que el área de la superficie de revolución resultante es infinita comparándola con la integral divergente $\int_{1}^{\infty} \frac{1}{x} dx$.
\end{enumerate}
\end{prob}

\begin{prob}[Función gamma] La función Gamma se define para $n > 0$ como: $\Gamma(n) = \int_{0}^{\infty} x^{n-1} e^{-x} \, dx$.
\begin{enumerate} 
  \item Use integración por partes para demostrar la relación de recurrencia $\Gamma(n+1) = n\Gamma(n)$.
  \item Demuestre que si $n$ es un entero positivo, $\Gamma(n) = (n-1)!$.
  \item Evalúe $\Gamma(1), \Gamma(2)$ y $\Gamma(3)$.
\end{enumerate}
\end{prob}

\begin{prob}[Convergencia de Intervalos Infinitos] Determine si la siguiente integral converge y, de ser así, encuentre su valor:
\[
\int_{-\infty}^{\infty} \frac{e^x}{1 + e^{2x}} \, dx
\]
\textit{Sugerencia: Divida la integral en dos intervalos $(-\infty, 0]$ y $[0, \infty)$ y use la sustitución $u = e^x$.}
\end{prob}
