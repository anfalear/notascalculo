\section{Razones de Cambio Relacionadas}
%%==========================================================

Si una cantidad $y$ depende del tiempo $t$, su derivada $\frac{dy}{dt}$
es su \textbf{razĆ³n de cambio respecto al tiempo}. Cuando dos o mĆ”s
cantidades estĆ”n ligadas por una ecuaciĆ³n geomĆ©trica o fĆ­sica, sus
derivadas temporales tambiĆ©n estĆ”n ligadas: son las llamadas
\textbf{razones de cambio relacionadas}. La clave es que la ecuaciĆ³n
que relaciona las cantidades vale para \emph{todo instante} $t$, por
lo que puede derivarse implĆ­citamente respecto a $t$.

\begin{pasos} Las razones de cambio relacionadas pueden resolverse siguiendo el siguiente protocolo:
\begin{enumerate}
    \item \textbf{Diagrama y variables.} Dibujar un esquema de la
    situaciĆ³n y asignar variables a las cantidades que cambian con $t$.
    \item \textbf{Identificar razones.} Listar las derivadas temporales
    conocidas y la que se desea encontrar.
    \item \textbf{EcuaciĆ³n fundamental.} Escribir una ecuaciĆ³n que
    relacione las variables y que sea vĆ”lida para todo $t$.
    \item \textbf{DerivaciĆ³n implĆ­cita.} Derivar ambos lados respecto
    a $t$ aplicando la regla de la cadena.
    \item \textbf{SustituciĆ³n y despeje.} Solo \emph{despuĆ©s} de derivar,
    sustituir los valores del instante particular y despejar la razĆ³n
    buscada.
\end{enumerate}
\end{pasos}

\begin{rem}
El error mĆ”s frecuente consiste en sustituir los valores numĆ©ricos
\emph{antes} de derivar. Al hacerlo, las variables se convierten en
constantes y sus derivadas desaparecen, destruyendo la informaciĆ³n
que se busca. Los valores numĆ©ricos solo entran en el paso 5.
\end{rem}

\begin{example}[Globo esfĆ©rico en inflado]
Un globo esfĆ©rico se infla a razĆ³n de $100\pi$ cm$^3$/s.

\textbf{Paso 2. Identificar razones.}
\begin{itemize}

\textbf{Paso 3. EcuaciĆ³n fundamental.}

\]

\textbf{Paso 4. DerivaciĆ³n implĆ­cita.}
Se deriva ambos lados respecto a $t$, aplicando la regla de la cadena:
\[
\frac{dV}{dt} = \frac{4}{3}\pi \cdot 3r^2 \cdot \frac{dr}{dt}

\textbf{Paso 5. SustituciĆ³n y despeje.}
Se sustituyen $r = 5$ cm y $\dfrac{dV}{dt} = 100\pi$ cm$^3$/s:

       = 4\pi \cdot 25\,\frac{dr}{dt}
       = 100\pi\,\frac{dr}{dt}.
\]
Despejando:
\[
\frac{dr}{dt} = \frac{100\pi}{100\pi} = \boxed{1 \text{ cm/s.}}
\]
\end{myproof}
\end{example}

\section{Valores Extremos}
%%==========================================================
Una de las aplicaciones mĆ”s importantes de la derivada es la localizaciĆ³n de los extremos de una funciĆ³n, a continuaciĆ³n su definiciĆ³n.

\begin{definition}
Sea $f$ definida en un conjunto $S$. Se dice que:
\begin{itemize}
    \item $f(c)$ es el \textbf{mĆ”ximo absoluto} de $f$ en $S$ si
    $f(c)\geq f(x)$ para todo $x\in S$.
    \item $f(c)$ es el \textbf{mĆ­nimo absoluto} de $f$ en $S$ si
    $f(c)\leq f(x)$ para todo $x\in S$.
    \end{itemize}
  \end{definition}
  \begin{definition} Se dice que $c$ es un \textbf{punto crĆ­tico} de $f$ si es un punto interior
    del dominio donde $f'(c)=0$ (\textbf{punto estacionario}) o donde
    $f'(c)$ no existe (\textbf{punto singular}).
\end{definition}


\begin{theorem}[Teorema del Valor Extremo de Weierstrass]
Si $f$ es continua en el intervalo cerrado $[a,b]$, entonces $f$ alcanza
un mĆ”ximo absoluto y un mĆ­nimo absoluto en $[a,b]$.
\end{theorem}

\begin{rem}
El Teorema de Weierstrass es un resultado de existencia que descansa
sobre la completitud de $\R$ y la compacidad de $[a,b]$: en un conjunto
abierto o en una funciĆ³n discontinua, el extremo puede no alcanzarse.
La funciĆ³n $f(x)=x$ en $(0,1)$ es el ejemplo elemental: estĆ” acotada
pero no tiene ni mĆ”ximo ni mĆ­nimo en el intervalo abierto.
\end{rem}

\begin{theorem}[Fermat]
Si $f$ tiene un extremo local en un punto interior $c$ del dominio, y
$f'(c)$ existe, entonces $f'(c)=0$.
\end{theorem}

\begin{pasos} El siguiente protocolo permite realizar la LocalizaciĆ³n de extremos absolutos en $\lbrack a,b\rbrack$
\begin{enumerate}
    \item Encontrar todos los puntos crĆ­ticos de $f$ en $(a,b)$.
    \item Evaluar $f$ en cada punto crĆ­tico y en los extremos $a$, $b$.
    \item El mayor valor es el mĆ”ximo absoluto; el menor, el mĆ­nimo
    absoluto.
\end{enumerate}
\end{pasos}

\begin{example}[Extremos absolutos de un polinomio cĆŗbico]
Encontrar los extremos absolutos de $f(x)=x^3-3x^2+1$ en
$\left[-\tfrac{1}{2},\,4\right]$.
\begin{myproof}
\textbf{Paso 1. Encontrar todos los puntos crĆ­ticos de $f$ en
$\left(-\tfrac{1}{2},\,4\right)$.}

Se calcula la derivada de $f$:
\[
f'(x) = 3x^2 - 6x = 3x(x-2).
\]
Como $f$ es un polinomio, $f'$ existe en todo punto, de modo que los
Ćŗnicos puntos crĆ­ticos son aquellos donde $f'(x)=0$:
\[
3x(x-2)=0 \implies x=0 \quad \text{o} \quad x=2.
\]
Ambos valores pertenecen al intervalo abierto
$\left(-\tfrac{1}{2},\,4\right)$, por lo que los dos son puntos crĆ­ticos
vĆ”lidos.

\textbf{Paso 2. Evaluar $f$ en cada punto crĆ­tico y en los extremos
del intervalo.}

Se evalĆŗa $f$ en los puntos crĆ­ticos $x=0$, $x=2$ y en los extremos
$x=-\tfrac{1}{2}$, $x=4$:

\begin{center}
\begin{tabular}{c l c}
\toprule
$x$ & \multicolumn{1}{c}{CĆ”lculo} & $f(x)$ \\
\midrule
$-\dfrac{1}{2}$ &
  $\left(-\tfrac{1}{2}\right)^3 - 3\left(-\tfrac{1}{2}\right)^2+1
   = -\tfrac{1}{8}-\tfrac{3}{4}+1
   = -\tfrac{1}{8}-\tfrac{6}{8}+\tfrac{8}{8}$ &
  $\dfrac{1}{8}$ \\[10pt]

$4$ & $64 - 48 + 1$ & $17$ \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Paso 3. Identificar el mĆ”ximo y el mĆ­nimo absolutos.}

Comparando los cuatro valores obtenidos,
\[

\]
se concluye:
\begin{itemize}
    \item \textbf{MĆ”ximo absoluto:} $f(4)=17$, alcanzado en el extremo
    derecho del intervalo.
    \item \textbf{MĆ­nimo absoluto:} $f(2)=-3$, alcanzado en el punto
    crĆ­tico interior $x=2$.
\end{itemize}
\[
\boxed{\text{MĆ”ximo absoluto: }f(4)=17;\quad\text{mĆ­nimo absoluto: }f(2)=-3.}
\]
\end{myproof}
\end{example}

En ocasiones, una funciĆ³n mantiene un comportamiento uniforme a lo largo
de un intervalo: siempre sube, siempre baja o permanece constante. Esta
regularidad se conoce como \textbf{monotonĆ­a} y constituye una de las
propiedades cualitativas mĆ”s importantes de una funciĆ³n, pues permite
anticipar su comportamiento global sin necesidad de evaluar cada punto.
El cĆ”lculo diferencial ofrece una herramienta precisa para detectarla:
el signo de la derivada. Si $f'(x)>0$, la funciĆ³n crece; si $f'(x)<0$,
decrece. Este vĆ­nculo entre la derivada y el comportamiento de $f$ queda
formalizado en el siguiente criterio.

\begin{theorem}[Criterio de monotonĆ­a]
Sea $f$ continua en $[a,b]$ y derivable en $(a,b)$:
\begin{itemize}
    \item Si $f'(x)>0$ para todo $x\in(a,b)$, entonces $f$ es
    \textbf{estrictamente creciente} en $[a,b]$.
    \item Si $f'(x)<0$ para todo $x\in(a,b)$, entonces $f$ es
    \textbf{estrictamente decreciente} en $[a,b]$.
    \item Si $f'(x)=0$ para todo $x\in(a,b)$, entonces $f$ es
    \textbf{constante} en $[a,b]$.
\end{itemize}
\end{theorem}

\begin{example}[MonotonĆ­a de $f(x)=x^3-3x$]
Determinar los intervalos de crecimiento y decrecimiento de
$f(x)=x^3-3x$.
\begin{myproof}
\textbf{Paso 1. Calcular $f'$ y encontrar los puntos crĆ­ticos.}

\[
f'(x) = 3x^2 - 3 = 3(x^2-1) = 3(x-1)(x+1).
\]
Los puntos crĆ­ticos se obtienen igualando $f'(x)=0$:
\[
3(x-1)(x+1)=0 \implies x=-1 \quad \text{o} \quad x=1.
\]
Estos dos valores dividen la recta real en tres intervalos:
$(-\infty,-1)$, $(-1,1)$ y $(1,+\infty)$.

\textbf{Paso 2. Analizar el signo de $f'$ en cada intervalo.}

\begin{center}
\begin{tabular}{c c c c}
\toprule
Intervalo & Signo de $(x+1)$ & Signo de $(x-1)$ & Signo de $f'(x)$ \\
\midrule
$(-\infty,-1)$ & $-$ & $-$ & $+$ \\
$(-1,\;1)$     & $+$ & $-$ & $-$ \\
$(1,+\infty)$  & $+$ & $+$ & $+$ \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Paso 3. Concluir la monotonĆ­a.}

\begin{itemize}
    \item $f'(x)>0$ en $(-\infty,-1)$: $f$ es \textbf{estrictamente
    creciente}.
    \item $f'(x)<0$ en $(-1,1)$: $f$ es \textbf{estrictamente
    decreciente}.
    \item $f'(x)>0$ en $(1,+\infty)$: $f$ es \textbf{estrictamente
    creciente}.
\end{itemize}

Los valores $f(-1)=2$ y $f(1)=-2$ corresponden respectivamente a un
mĆ”ximo local y un mĆ­nimo local de $f$.
\[
\boxed{f\text{ crece en }(-\infty,-1)\cup(1,+\infty);\quad f\text{ decrece en }(-1,1).}
\]

La \textbf{concavidad} describe hacia quĆ© lado Ā«abreĀ» la curva de $f$
en cada intervalo: como una taza ($\smile$) o como una cĆŗpula ($\frown$).
GeomĆ©tricamente, una figura es \emph{convexa} si el segmento que une
cualquier par de puntos interiores permanece dentro de ella, y
\emph{cĆ³ncava} en caso contrario. En el anĆ”lisis de funciones, este
comportamiento lo detecta la segunda derivada: su signo indica si la
pendiente $f'$ crece o decrece a medida que $x$ avanza.

\begin{definition}
La grĆ”fica de $f$ es \textbf{cĆ³ncava hacia arriba} en $(a,b)$ si $f'$
es creciente en ese intervalo, y \textbf{cĆ³ncava hacia abajo} si $f'$
es decreciente. Equivalentemente:
\[
f''(x)>0 \implies \text{cĆ³ncava hacia arriba}, \qquad
f''(x)<0 \implies \text{cĆ³ncava hacia abajo}.
\]
Un punto $c$ es un \textbf{punto de inflexiĆ³n} de $f$ si $f$ es continua
en $c$ y la concavidad cambia de signo en $c$.
\end{definition}

\begin{rem}
Un punto donde $f''(c)=0$ no es necesariamente un punto de inflexiĆ³n:
se requiere que $f''$ \emph{cambie de signo} en $c$. La funciĆ³n
$f(x)=x^4$ satisface $f''(0)=0$, pero la concavidad no cambia en $x=0$
(es cĆ³ncava hacia arriba en todo $\R$), por lo que $x=0$ no es punto de
inflexiĆ³n.
\end{rem}

Las herramientas desarrolladas hasta ahora de monotonĆ­a y concavidad
permiten caracterizar puntos especiales que son mĆ”ximos o mĆ­nimos, no
en el dominio completo de $f$, sino en una vecindad reducida alrededor
de un punto. Estos se denominan \textbf{extremos locales} y representan
los \textit{picos} y \textit{valles} de la grĆ”fica. Su localizaciĆ³n se apoya en los
puntos crĆ­ticos y puede abordarse mediante dos criterios: uno basado en
el cambio de signo de $f'$ y otro, mĆ”s Ć”gil cuando es aplicable, basado
en el signo de $f''$.

\begin{definition}
$f(c)$ es un \textbf{mĆ”ximo local} si existe un intervalo abierto
$(a,b)$ que contiene a $c$ tal que $f(c)\geq f(x)$ para todo
$x\in(a,b)$. AnĆ”logamente se define el \textbf{mĆ­nimo local}.
\end{definition}

\begin{theorem}[Criterio de la primera derivada]
Sea $c$ un punto crĆ­tico de $f$:
\begin{itemize}
    \item Si $f'$ cambia de $+$ a $-$ en $c$: \textbf{mĆ”ximo local}.
    \item Si $f'$ cambia de $-$ a $+$ en $c$: \textbf{mĆ­nimo local}.
    \item Si $f'$ no cambia de signo en $c$: no hay extremo local.
\end{itemize}
\end{theorem}

\begin{theorem}[Criterio de la segunda derivada]
Si $f'(c)=0$ y $f''(c)$ existe:
\begin{itemize}
    \item $f''(c)<0$: \textbf{mĆ”ximo local} (la curva es cĆ³ncava hacia
    abajo en $c$).
    \item $f''(c)>0$: \textbf{mĆ­nimo local} (la curva es cĆ³ncava hacia
    arriba en $c$).
    \item $f''(c)=0$: el criterio es \textbf{inconcluso}; puede haber
    mĆ”ximo, mĆ­nimo o punto de inflexiĆ³n.
\end{itemize}
\end{theorem}

\begin{rem}
El caso $f''(c)=0$ ilustra la importancia de no abandonar el anĆ”lisis
prematuramente. Para $f(x)=x^4$, se tiene $f'(0)=0$ y $f''(0)=0$,
pero $x=0$ es un mĆ­nimo local; para $f(x)=x^3$, las mismas condiciones
se cumplen en $x=0$, pero se trata de un punto de inflexiĆ³n. El criterio
de la segunda derivada no distingue estos casos, por lo que siempre debe
recurrirse al de la primera derivada cuando $f''(c)=0$.
\end{rem}

\section{Trazado SistemĆ”tico de Curvas}
%%==========================================================

El anĆ”lisis desarrollado en las secciones anteriores no es un fin en sĆ­
mismo: monotonĆ­a, concavidad, extremos locales y puntos de inflexiĆ³n son
piezas de un mismo rompecabezas cuyo resultado final es la comprensiĆ³n
global del comportamiento de una funciĆ³n. Reunir todas estas herramientas
en un procedimiento ordenado permite \textbf{bosquejar la grĆ”fica de
cualquier funciĆ³n} de manera rigurosa, sin depender de tablas de valores
construidas punto a punto. La ventaja de este enfoque es doble: por un
lado, la grĆ”fica resultante refleja fielmente la estructura matemĆ”tica de
$f$; por el otro, el proceso mismo de anĆ”lisis obliga a comprender la
funciĆ³n antes de representarla. El protocolo que se presenta a
continuaciĆ³n organiza ese proceso en cuatro etapas progresivas, donde
cada paso alimenta al siguiente.

\begin{pasos}
Para bosquejar la grĆ”fica de una funciĆ³n es posible seguir el siguiente
proceso sistemĆ”tico:
\begin{enumerate}
    \item \textbf{AnĆ”lisis previo.} Determinar dominio, rango (si es
    posible), simetrĆ­a (par/impar/periĆ³dica), intersecciones con los
    ejes y comportamiento asintĆ³tico ($x\to\pm\infty$, puntos
    singulares).
    \item \textbf{Primera derivada.} Calcular $f'$, hallar puntos
    crĆ­ticos, determinar intervalos de crecimiento y decrecimiento,
    clasificar extremos locales.
    \item \textbf{Segunda derivada.} Calcular $f''$, determinar
    intervalos de concavidad y localizar puntos de inflexiĆ³n.
    \item \textbf{Tabla y bosquejo.} Resumir toda la informaciĆ³n en
    una tabla de puntos clave. Trazar primero las asĆ­ntotas, luego
    los puntos especiales, y conectarlos respetando la monotonĆ­a y la
    concavidad.
\end{enumerate}
\end{pasos}

\begin{example}[Trazado de $f(x)=x^2/(x^2-4)$]
Trazar la curva $f(x)=\dfrac{x^2}{x^2-4}$.
\begin{myproof}
\textbf{Paso 1. AnĆ”lisis previo.}

\emph{Dominio.} El denominador $x^2-4=(x-2)(x+2)$ se anula en $x=\pm2$,
de modo que el dominio es $\mathbb{R}\setminus\{-2,\,2\}$.

\emph{SimetrĆ­a.} Como $f(-x)=\dfrac{(-x)^2}{(-x)^2-4}=\dfrac{x^2}{x^2-4}=f(x)$,
la funciĆ³n es \textbf{par}: su grĆ”fica es simĆ©trica respecto al eje $y$.
Esto permite analizar $x\geq 0$ y reflejar.

\emph{Intersecciones.} En $x=0$: $f(0)=0$, luego la curva pasa por el
origen. La ecuaciĆ³n $f(x)=0$ implica $x^2=0$, es decir, $x=0$ es la
Ćŗnica intersecciĆ³n con ambos ejes.

\emph{AsĆ­ntotas verticales.} En $x=2$ y $x=-2$ el denominador se anula
y el numerador no, por lo que existen asĆ­ntotas verticales en ambos
puntos. Analizando los lĆ­mites laterales en $x=2$:
\[
\lim_{x\to 2^-}f(x)=-\infty, \qquad \lim_{x\to 2^+}f(x)=+\infty,
\]
y por simetrĆ­a los comportamientos en $x=-2$ son opuestos.

\emph{AsĆ­ntota horizontal.} Dividiendo numerador y denominador entre
$x^2$:
\[
\lim_{x\to\pm\infty}f(x)=\lim_{x\to\pm\infty}\frac{1}{1-4/x^2}=1,
\]
de modo que $y=1$ es asĆ­ntota horizontal. NĆ³tese ademĆ”s que $f(x)=1$
no tiene soluciĆ³n ($x^2=x^2-4$ es imposible), asĆ­ que la curva nunca
toca su propia asĆ­ntota horizontal.

\textbf{Paso 2. Primera derivada: monotonĆ­a y extremos locales.}

Aplicando la regla del cociente:
\[

= \frac{2x(x^2-4)-x^2\cdot 2x}{(x^2-4)^2}

determinado Ćŗnicamente por $-8x$:

\begin{center}
\begin{tabular}{c c c}
\toprule
Intervalo & Signo de $f'(x)$ & MonotonĆ­a \\
\midrule
$(-\infty,-2)$ & $+$ & creciente \\
$(-2,\;0)$     & $+$ & creciente \\
$(0,\;2)$      & $-$ & decreciente \\
$(2,+\infty)$  & $-$ & decreciente \\
\bottomrule
\end{tabular}
\end{center}

En $x=0$ la derivada cambia de $+$ a $-$, lo que indica un
\textbf{mĆ”ximo local} con valor $f(0)=0$. ObsĆ©rvese que, aunque la
funciĆ³n crece en $(-\infty,-2)$ y en $(-2,0)$, estos son intervalos
separados por la discontinuidad en $x=-2$; no se puede afirmar que $f$
crece en $(-\infty,0)$ en su conjunto.

\textbf{Paso 3. Segunda derivada: concavidad y puntos de inflexiĆ³n.}

Derivando $f'(x)=\dfrac{-8x}{(x^2-4)^2}$ mediante la regla del cociente:
\[

= \frac{-8(x^2-4)^2 - (-8x)\cdot 2(x^2-4)\cdot 2x}{(x^2-4)^4}
= \frac{-8(x^2-4)+32x^2}{(x^2-4)^3}

\]
El numerador $8(3x^2+4)>0$ para todo $x\in\mathbb{R}$, de modo que el
signo de $f''$ depende exclusivamente del denominador $(x^2-4)^3$:

\begin{center}
\begin{tabular}{c c c c}
\toprule
Intervalo & Signo de $(x^2-4)^3$ & Signo de $f''(x)$ & Concavidad \\
\midrule
$(-\infty,-2)$ & $+$ & $+$ & hacia arriba $\smile$ \\
$(-2,\;2)$     & $-$ & $-$ & hacia abajo  $\frown$ \\
$(2,+\infty)$  & $+$ & $+$ & hacia arriba $\smile$ \\
\bottomrule
\end{tabular}
\end{center}

Como $f''(x)=0$ no tiene soluciĆ³n real, \textbf{no hay puntos de
inflexiĆ³n}. La concavidad cambia Ćŗnicamente en las asĆ­ntotas verticales,
que no pertenecen al dominio.

\midrule

$(-2,\;0)$     & $+$ & $-$ & creciente, cĆ³ncava $\downarrow$ \\
$(0,\;2)$      & $-$ & $-$ & decreciente, cĆ³ncava $\downarrow$ \\

\section{El Teorema del Valor Medio}
%%==========================================================

Los criterios de monotonĆ­a y optimizaciĆ³n desarrollados en las secciones
anteriores descansan sobre una garantĆ­a mĆ”s profunda: si una funciĆ³n es
suficientemente regular en un intervalo, su derivada no puede comportarse
de forma arbitraria. El resultado que formaliza esta idea es el
\textbf{Teorema del Valor Medio}, uno de los teoremas mĆ”s importantes
del cĆ”lculo diferencial. Su punto de partida es un caso particular
elemental, conocido como el Lema de Rolle, que captura una observaciĆ³n
geomĆ©trica sencilla: si una curva comienza y termina a la misma altura,
en algĆŗn punto intermedio debe ser horizontal.

\begin{lemma}[Rolle]
Si $f$ es continua en $[a,b]$, derivable en $(a,b)$ y $f(a)=f(b)$,
entonces existe al menos un $c\in(a,b)$ tal que $f'(c)=0$.
\end{lemma}

\begin{proof}[Esquema]
Por el Teorema del Valor Extremo de Weierstrass, $f$ alcanza en $[a,b]$ un
mĆ”ximo y un mĆ­nimo absolutos. Si ambos se alcanzan en los extremos, entonces
$f$ es constante (pues $f(a)=f(b)$) y $f'(c)=0$ para todo $c\in(a,b)$. En caso
contrario, al menos uno de los dos extremos absolutos se alcanza en un punto
interior $c\in(a,b)$; siendo $f$ derivable allĆ­, el Teorema de Fermat garantiza
que $f'(c)=0$.
\end{proof}

El Teorema del Valor Medio generaliza esta idea eliminando la exigencia
de que los extremos tengan la misma imagen: ahora se garantiza la
existencia de un punto donde la pendiente de la tangente coincide con
la pendiente de la recta que une los dos extremos de la curva.

\begin{theorem}[del Valor Medio de Lagrange]
Si $f$ es continua en $[a,b]$ y derivable en $(a,b)$, existe al menos
un $c\in(a,b)$ tal que
\[
f'(c) = \frac{f(b)-f(a)}{b-a}.
\]
\end{theorem}

GeomĆ©tricamente, el TVM garantiza que existe un punto en la curva donde
la recta tangente es \textbf{paralela} a la recta secante que une los
extremos del intervalo. La interpretaciĆ³n fĆ­sica es igualmente
iluminadora: si un automĆ³vil recorre $120$ km en $2$ horas (velocidad
promedio $60$ km/h), en algĆŗn instante exacto su velocidad instantĆ”nea
fue exactamente $60$ km/h. El siguiente ejemplo muestra que el TVM no
es un resultado independiente, sino una consecuencia directa del Lema
de Rolle.

Al calcular lĆ­mites, es frecuente encontrar expresiones donde numerador

\begin{theorem}[Regla de L'HĆ“pital]
SupĆ³ngase que $\lim_{x\to a}f(x)=0$ y $\lim_{x\to a}g(x)=0$ tiene forma

$g'(x)\neq 0$ allĆ­, y el lĆ­mite $\lim_{x\to a}\frac{f'(x)}{g'(x)}$
existe (o es $\pm\infty$), entonces

\]
El resultado tambiĆ©n vale para $a=\pm\infty$ y para lĆ­mites laterales.
\end{theorem}

no aplicar la regla del cociente. Dos errores frecuentes merecen
atenciĆ³n especial: el primero es aplicarla cuando el lĆ­mite \emph{no} es una
forma indeterminada, lo que produce resultados incorrectos; y el segundo aplicarla
repetidamente sin verificar que el nuevo cociente siga siendo
indeterminado. Si tras una aplicaciĆ³n el lĆ­mite permanece indeterminado,
la regla puede reiterarse; si el cociente de derivadas oscila sin

\end{rem}

\section{Diferenciales y Aproximaciones Lineales}
%%==========================================================

Cuando $f'(a)$ existe, la recta tangente en $(a,f(a))$ es la mejor
aproximaciĆ³n lineal de $f$ cerca de $a$. Esta idea se formaliza

\begin{definition}

\[

donde $dx$ es un incremento arbitrario en $x$. El diferencial $df$

\end{definition}

\begin{rem}

es del orden de $(dx)^2$ para funciones suaves:
$\Delta y = dy + \mathcal{O}\!\left((dx)^2\right)$.
Cuando $|dx|\ll 1$, el error relativo $(\Delta y - dy)/\Delta y$
tambiĆ©n es pequeĆ±o, por lo que $dy$ es una aproximaciĆ³n prĆ”ctica Ćŗtil.

\begin{theorem}[AproximaciĆ³n lineal o linealizaciĆ³n]

\[

\]
$L(x)$ se denomina la \textbf{linealizaciĆ³n de $f$ en $a$}; su grĆ”fica
es la recta tangente a $f$ en el punto $(a,f(a))$.
\end{theorem}

\begin{pasos}
Para estimar $f(b)$ cuando $b$ estĆ” cerca de un punto $a$ conveniente:
\begin{enumerate}
    \item Elegir $a$ cercano a $b$ donde $f(a)$ sea exactamente
    calculable.
    \item Calcular $f'(a)$.
    \item Aplicar: $f(b) \approx f(a) + f'(a)(b-a)$.
\end{enumerate}
\end{pasos}

\begin{example}[AproximaciĆ³n lineal de $\sqrt{4.02}$]
Usar la aproximaciĆ³n lineal para estimar $\sqrt{4.02}$.
\begin{myproof}
\textbf{Paso 1. Elegir el punto base.}

Se toma $f(x)=\sqrt{x}$ y $a=4$, pues $f(4)=\sqrt{4}=2$ es exacto.

\textbf{Paso 2. Calcular $f'(a)$.}

\[
f'(x) = \frac{1}{2\sqrt{x}} \implies f'(4) = \frac{1}{4}.
\]

\textbf{Paso 3. Aplicar la aproximaciĆ³n lineal.}

Con $b=4.02$ y $b-a=0.02$:
\[
\sqrt{4.02} \approx f(4) + f'(4)(4.02-4)
= 2 + \frac{1}{4}(0.02) = \boxed{2.005}.
\]
El valor exacto es $\sqrt{4.02}\approx 2.00499\ldots$; el error
absoluto es menor que $0.00001$.
\end{myproof}
\end{example}

\begin{example}[AproximaciĆ³n lineal de $\sen 31^\circ$]
Usar la aproximaciĆ³n lineal para estimar $\sen 31^\circ$.
\begin{myproof}
\textbf{Paso 1. Elegir el punto base.}

Se trabaja en radianes: $31^\circ = \dfrac{31\pi}{180}$ rad, cercano
a $a=\dfrac{\pi}{6}$ ($=30^\circ$), para el que $\sen\dfrac{\pi}{6}=\dfrac{1}{2}$
es exacto.

\textbf{Paso 2. Calcular $f'(a)$.}

Con $f(x)=\sen x$:
\[
f'(x)=\cos x \implies f'\!\left(\frac{\pi}{6}\right)=\cos\frac{\pi}{6}
=\frac{\sqrt{3}}{2}.
\]

\textbf{Paso 3. Aplicar la aproximaciĆ³n lineal.}

El incremento es $\Delta x=\dfrac{\pi}{180}$ rad ($=1^\circ$):
\[
\sen 31^\circ \approx \frac{1}{2}+\frac{\sqrt{3}}{2}\cdot\frac{\pi}{180}
\approx 0.5000 + 0.0151 = \boxed{0.5151}.
\]
El valor exacto es $\sen 31^\circ\approx 0.5150\ldots$; error menor que $0.0001$.
\end{myproof}
\end{example}
