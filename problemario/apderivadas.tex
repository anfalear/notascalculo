\section{Panorama General}
%%==========================================================

La derivada no es Ășnicamente un objeto teĂłrico: es el instrumento
central para analizar el comportamiento de funciones y resolver
problemas cuantitativos en fĂ­sica, ingenierĂ­a, economĂ­a y geometrĂ­a.
Este capĂ­tulo desarrolla las aplicaciones mĂĄs importantes, organizadas
en un orden pedagĂłgico natural: primero el anĂĄlisis del comportamiento
local (extremos, monotonĂ­a, concavidad), luego las aplicaciones al
trazado sistemĂĄtico de curvas y a la optimizaciĂłn, y finalmente las
herramientas analĂ­ticas transversales (Teorema del Valor Medio, regla
de L'HĂŽpital) y numĂ©ricas (Newton-Raphson, bisecciĂłn).

\begin{rem}
El hilo conductor de este capĂ­tulo es el siguiente principio: la
derivada mide la \textbf{tasa de variaciĂłn instantĂĄnea}. SegĂșn el
contexto, esta idea adquiere formas distintas ---velocidad, pendiente,
crecimiento marginal, tasa de llenado--- pero el aparato matemĂĄtico
es siempre el mismo. Reconocer esta unidad subyacente facilita la
transferencia de las tĂ©cnicas entre contextos aparentemente dispares.
\end{rem}

\section{Razones de Cambio Relacionadas}
%%==========================================================

\subsection{Fundamento}

Si una cantidad $y$ depende del tiempo $t$, su derivada $\frac{dy}{dt}$
es su \textbf{razĂłn de cambio respecto al tiempo}. Cuando dos o mĂĄs
cantidades estĂĄn ligadas por una ecuaciĂłn geomĂ©trica o fĂ­sica, sus
derivadas temporales tambiĂ©n estĂĄn ligadas: son las llamadas
\textbf{razones de cambio relacionadas}. La clave es que la ecuaciĂłn
que relaciona las cantidades vale para \emph{todo instante} $t$, por
lo que puede derivarse implĂ­citamente respecto a $t$.

\begin{pasos}
\begin{enumerate}
    \item \textbf{Diagrama y variables.} Dibujar un esquema de la
    situaciĂłn y asignar variables a las cantidades que cambian con $t$.
    \item \textbf{Identificar razones.} Listar las derivadas temporales
    conocidas y la que se desea encontrar.
    \item \textbf{EcuaciĂłn fundamental.} Escribir una ecuaciĂłn que
    relacione las variables y que sea vĂĄlida para todo $t$.
    \item \textbf{DerivaciĂłn implĂ­cita.} Derivar ambos lados respecto
    a $t$ aplicando la regla de la cadena.
    \item \textbf{SustituciĂłn y despeje.} Solo \emph{despuĂ©s} de derivar,
    sustituir los valores del instante particular y despejar la razĂłn
    buscada.
\end{enumerate}
\end{pasos}

\begin{rem}
El error mĂĄs frecuente consiste en sustituir los valores numĂ©ricos
\emph{antes} de derivar. Al hacerlo, las variables se convierten en
constantes y sus derivadas desaparecen, destruyendo la informaciĂłn
que se busca. Los valores numĂ©ricos solo entran en el paso 5.
\end{rem}

\subsection{Ejemplos}

\begin{example}
Un tanque cĂłnico invertido tiene altura $12$ m y radio basal $6$ m.
El agua entra a razĂłn de $2$ m$^3$/min. ÂżA quĂ© rapidez sube el nivel
cuando el agua tiene $4$ m de profundidad?
\end{example}

\begin{myproof}
\textbf{Paso 1.} Variables: $h$ = profundidad del agua, $r$ = radio
de la superficie libre, $V$ = volumen. Se conoce $\frac{dV}{dt}=2$
m$^3$/min y se busca $\frac{dh}{dt}$ cuando $h=4$.

\textbf{Paso 3.} Por semejanza de triĂĄngulos,
$\frac{r}{h}=\frac{6}{12}=\frac{1}{2}$, de modo que $r=\frac{h}{2}$.
Sustituyendo en la fĂłrmula del cono:
\[
V = \frac{1}{3}\pi r^2 h = \frac{1}{3}\pi\left(\frac{h}{2}\right)^2 h
= \frac{\pi h^3}{12}.
\]

\textbf{Paso 4.} Derivando respecto a $t$:
\[
\frac{dV}{dt} = \frac{\pi h^2}{4}\cdot\frac{dh}{dt}.
\]

\textbf{Paso 5.} Con $h=4$ y $\frac{dV}{dt}=2$:
\[
2 = \frac{\pi(16)}{4}\cdot\frac{dh}{dt}
\implies \frac{dh}{dt} = \frac{2}{4\pi} = \frac{1}{2\pi}
\approx 0.159 \text{ m/min}. \qquad\square
\]
\end{myproof}

\begin{example}
Una escalera de $10$ m de largo se apoya en una pared vertical. La base
se aleja de la pared a $0.5$ m/s. ÂżA quĂ© velocidad desciende el extremo
superior cuando la base estĂĄ a $6$ m de la pared?
\end{example}

\begin{myproof}
Sea $x$ la distancia de la base a la pared y $y$ la altura del extremo
superior. La restricciĂłn es $x^2+y^2=100$.

Derivando respecto a $t$:
\[
2x\frac{dx}{dt}+2y\frac{dy}{dt}=0
\implies \frac{dy}{dt} = -\frac{x}{y}\cdot\frac{dx}{dt}.
\]
Cuando $x=6$: $y=\sqrt{100-36}=8$. Con $\frac{dx}{dt}=0.5$:
\[
\frac{dy}{dt} = -\frac{6}{8}\cdot 0.5 = -\frac{3}{8} \text{ m/s}.
\]
El signo negativo indica que el extremo superior \emph{desciende} a
$\frac{3}{8}$ m/s. $\square$
\end{myproof}

\begin{example}
Un globo esfĂ©rico se infla a razĂłn de $100\pi$ cm$^3$/s.
ÂżA quĂ© rapidez crece el radio cuando el radio mide $5$ cm?
\end{example}

\begin{myproof}
$V=\frac{4}{3}\pi r^3$. Derivando: $\frac{dV}{dt}=4\pi r^2\frac{dr}{dt}$.
Con $r=5$ y $\frac{dV}{dt}=100\pi$:
\[
100\pi = 4\pi(25)\frac{dr}{dt}
\implies \frac{dr}{dt} = 1 \text{ cm/s}. \qquad\square
\]
\end{myproof}

\section{Valores Extremos}
%%==========================================================

\subsection{Extremos absolutos y puntos crĂ­ticos}

\begin{definition}
Sea $f$ definida en un conjunto $S$. Se dice que:
\begin{itemize}
    \item $f(c)$ es el \textbf{mĂĄximo absoluto} de $f$ en $S$ si
    $f(c)\geq f(x)$ para todo $x\in S$.
    \item $f(c)$ es el \textbf{mĂ­nimo absoluto} de $f$ en $S$ si
    $f(c)\leq f(x)$ para todo $x\in S$.
    \item $c$ es un \textbf{punto crĂ­tico} de $f$ si es un punto interior
    del dominio donde $f'(c)=0$ (\textbf{punto estacionario}) o donde
    $f'(c)$ no existe (\textbf{punto singular}).
\end{itemize}
\end{definition}

\begin{theorem}[Weierstrass --- Teorema del Valor Extremo]
Si $f$ es continua en el intervalo cerrado $[a,b]$, entonces $f$ alcanza
un mĂĄximo absoluto y un mĂ­nimo absoluto en $[a,b]$.
\end{theorem}

\begin{rem}
El Teorema de Weierstrass es un resultado de existencia que descansa
sobre la completitud de $\R$ y la compacidad de $[a,b]$: en un conjunto
abierto o en una funciĂłn discontinua, el extremo puede no alcanzarse.
La funciĂłn $f(x)=x$ en $(0,1)$ es el ejemplo elemental: estĂĄ acotada
pero no tiene ni mĂĄximo ni mĂ­nimo en el intervalo abierto.
\end{rem}

\begin{theorem}[Fermat]
Si $f$ tiene un extremo local en un punto interior $c$ del dominio, y
$f'(c)$ existe, entonces $f'(c)=0$.
\end{theorem}

\begin{pasos}[LocalizaciĂłn de extremos absolutos en $\lbrack a,b\rbrack$]
\begin{enumerate}
    \item Encontrar todos los puntos crĂ­ticos de $f$ en $(a,b)$.
    \item Evaluar $f$ en cada punto crĂ­tico y en los extremos $a$, $b$.
    \item El mayor valor es el mĂĄximo absoluto; el menor, el mĂ­nimo
    absoluto.
\end{enumerate}
\end{pasos}

\begin{example}
Encontrar los extremos absolutos de $f(x)=x^3-3x^2+1$ en $[-\tfrac{1}{2},4]$.
\end{example}

\begin{myproof}
$f'(x)=3x^2-6x=3x(x-2)$. Puntos crĂ­ticos en el interior: $x=0$ y $x=2$.
Evaluamos $f$ en los puntos crĂ­ticos y extremos del intervalo:
\begin{center}
\begin{tabular}{c c}
\toprule
$x$ & $f(x)$ \\
\midrule
$-\frac{1}{2}$ & $\frac{1}{8}+\frac{3}{4}+1 = \frac{15}{8}$ \\
$0$ & $1$ \\
$2$ & $8-12+1=-3$ \\
$4$ & $64-48+1=17$ \\
\bottomrule
\end{tabular}
\end{center}
MĂĄximo absoluto: $f(4)=17$. MĂ­nimo absoluto: $f(2)=-3$. $\square$
\end{myproof}

\section{MonotonĂ­a y Concavidad}
%%==========================================================

\subsection{Intervalos de crecimiento y decrecimiento}

\begin{theorem}[Criterio de monotonĂ­a]
Sea $f$ continua en $[a,b]$ y derivable en $(a,b)$:
\begin{itemize}
    \item Si $f'(x)>0$ para todo $x\in(a,b)$, entonces $f$ es
    \textbf{estrictamente creciente} en $[a,b]$.
    \item Si $f'(x)<0$ para todo $x\in(a,b)$, entonces $f$ es
    \textbf{estrictamente decreciente} en $[a,b]$.
    \item Si $f'(x)=0$ para todo $x\in(a,b)$, entonces $f$ es
    \textbf{constante} en $[a,b]$.
\end{itemize}
\end{theorem}

\subsection{Concavidad y puntos de inflexiĂłn}

\begin{definition}
La grĂĄfica de $f$ es \textbf{cĂłncava hacia arriba} en $(a,b)$ si $f'$
es creciente en ese intervalo, y \textbf{cĂłncava hacia abajo} si $f'$
es decreciente. Equivalentemente:
\[
f''(x)>0 \implies \text{cĂłncava hacia arriba}, \qquad
f''(x)<0 \implies \text{cĂłncava hacia abajo}.
\]
Un punto $c$ es un \textbf{punto de inflexiĂłn} de $f$ si $f$ es continua
en $c$ y la concavidad cambia de signo en $c$.
\end{definition}

\begin{rem}
Un punto donde $f''(c)=0$ no es necesariamente un punto de inflexiĂłn:
se requiere que $f''$ \emph{cambie de signo} en $c$. La funciĂłn
$f(x)=x^4$ satisface $f''(0)=0$, pero la concavidad no cambia en $x=0$
(es cĂłncava hacia arriba en todo $\R$), por lo que $x=0$ no es punto de
inflexiĂłn.
\end{rem}

\section{Extremos Locales}
%%==========================================================

\begin{definition}
$f(c)$ es un \textbf{mĂĄximo local} si existe un intervalo abierto
$(a,b)$ que contiene a $c$ tal que $f(c)\geq f(x)$ para todo
$x\in(a,b)$. AnĂĄlogamente se define el \textbf{mĂ­nimo local}.
\end{definition}

\begin{theorem}[Criterio de la primera derivada]
Sea $c$ un punto crĂ­tico de $f$:
\begin{itemize}
    \item Si $f'$ cambia de $+$ a $-$ en $c$: \textbf{mĂĄximo local}.
    \item Si $f'$ cambia de $-$ a $+$ en $c$: \textbf{mĂ­nimo local}.
    \item Si $f'$ no cambia de signo en $c$: no hay extremo local.
\end{itemize}
\end{theorem}

\begin{theorem}[Criterio de la segunda derivada]
Si $f'(c)=0$ y $f''(c)$ existe:
\begin{itemize}
    \item $f''(c)<0$: \textbf{mĂĄximo local}.
    \item $f''(c)>0$: \textbf{mĂ­nimo local}.
    \item $f''(c)=0$: el criterio es \textbf{inconcluso}; puede haber
    mĂĄximo, mĂ­nimo o punto de inflexiĂłn.
\end{itemize}
\end{theorem}

\begin{rem}
El caso $f''(c)=0$ ilustra la importancia de no abandonar el anĂĄlisis
prematuramente. Para $f(x)=x^4$, $f'(0)=0$ y $f''(0)=0$, pero $x=0$
es un mĂ­nimo local (verificable con el criterio de la primera derivada).
Para $f(x)=x^3$, $f'(0)=0$ y $f''(0)=0$, pero $x=0$ es un punto de
inflexiĂłn. El criterio de la segunda derivada no distingue estos casos;
siempre se debe recurrir al de la primera derivada cuando la segunda es
cero.
\end{rem}

\begin{example}
Analizar completamente $f(x)=x^4-4x^3$.
\end{example}

\begin{myproof}
$f'(x)=4x^3-12x^2=4x^2(x-3)$. Puntos crĂ­ticos: $x=0$ y $x=3$.

$f''(x)=12x^2-24x=12x(x-2)$. En $x=3$: $f''(3)=36>0$ $\Rightarrow$
mĂ­nimo local $f(3)=-27$. En $x=0$: $f''(0)=0$, inconcluso.
Analizando signo de $f'$: para $x<0$, $f'<0$; para $0<x<3$, $f'<0$;
para $x>3$, $f'>0$. Como $f'$ no cambia de signo en $x=0$, no hay
extremo local allĂ­.

Puntos de inflexiĂłn ($f''=0$): $x=0$ y $x=2$.
En $x=0$: $f''$ cambia de $+$ a $-$... verificar: $f''(-1)=36>0$,
$f''(1)=-12<0$. SĂ­ cambia $\Rightarrow$ inflexiĂłn en $(0,0)$.
En $x=2$: $f''(1)=-12<0$, $f''(3)=36>0$. Cambia $\Rightarrow$
inflexiĂłn en $(2,-16)$. $\square$
\end{myproof}

\section{Trazado SistemĂĄtico de Curvas}
%%==========================================================

El anĂĄlisis completo de una funciĂłn antes de graficarla integra todas
las herramientas previas en un procedimiento organizado.

\begin{pasos}
\begin{enumerate}
    \item \textbf{AnĂĄlisis previo.} Determinar dominio, rango (si es
    posible), simetrĂ­a (par/impar/periĂłdica), intersecciones con los
    ejes y comportamiento asintĂłtico ($x\to\pm\infty$, puntos
    singulares).
    \item \textbf{Primera derivada.} Calcular $f'$, hallar puntos
    crĂ­ticos, determinar intervalos de crecimiento y decrecimiento,
    clasificar extremos locales.
    \item \textbf{Segunda derivada.} Calcular $f''$, determinar
    intervalos de concavidad y localizar puntos de inflexiĂłn.
    \item \textbf{Tabla y bosquejo.} Resumir toda la informaciĂłn en
    una tabla de puntos clave. Trazar primero las asĂ­ntotas, luego
    los puntos especiales, y conectarlos respetando la monotonĂ­a y la
    concavidad.
\end{enumerate}
\end{pasos}

\begin{example}
Trazar la curva $f(x)=\dfrac{x^2}{x^2-4}$.
\end{example}

\begin{myproof}
\textbf{Paso 1.}
\emph{Dominio:} $\R\setminus\{-2,2\}$.
\emph{SimetrĂ­a:} $f(-x)=f(x)$, la funciĂłn es par.
\emph{Intersecciones:} $f(0)=0$; $f(x)=0\Rightarrow x=0$.
\emph{AsĂ­ntotas verticales:} $x=\pm 2$ (denominador se anula).
\emph{AsĂ­ntota horizontal:} $\lim_{x\to\infty}f(x)=1$, i.e., $y=1$.

\textbf{Paso 2.}
\[
f'(x) = \frac{2x(x^2-4)-x^2\cdot 2x}{(x^2-4)^2}
= \frac{-8x}{(x^2-4)^2}.
\]
$f'(x)=0\Rightarrow x=0$; $f''(0)=?$ Signo de $f'$: para $x<0$,
$f'>0$ (creciente); para $x>0$, $f'<0$ (decreciente). MĂĄximo local
en $x=0$: $f(0)=0$.

\textbf{Paso 3.}
\[
f''(x) = \frac{-8(x^2-4)^2-(-8x)\cdot 2(x^2-4)\cdot 2x}{(x^2-4)^4}
= \frac{8(3x^2+4)}{(x^2-4)^3}.
\]
$f''(x)=0$ no tiene soluciĂłn real ($3x^2+4>0$ siempre). En
$(-2,2)$: denominador $<0$ y numerador $>0$, luego $f''<0$ (cĂłncava
hacia abajo). Fuera de $(-2,2)$: $f''>0$ (cĂłncava hacia arriba).
No hay puntos de inflexiĂłn.

\textbf{Resumen:}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c c c c}
\toprule
Intervalo & $f'$ & $f''$ & Comportamiento \\
\midrule
$(-\infty,-2)$ & $+$ & $+$ & creciente, cĂłncava $\uparrow$ \\
$(-2,0)$ & $+$ & $-$ & creciente, cĂłncava $\downarrow$ \\
$(0,2)$ & $-$ & $-$ & decreciente, cĂłncava $\downarrow$ \\
$(2,+\infty)$ & $-$ & $+$ & decreciente, cĂłncava $\uparrow$ \\
\bottomrule
\end{tabular}
\end{center}
$\square$
\end{myproof}

\section{OptimizaciĂłn}
%%==========================================================

\subsection{MetodologĂ­a}

La optimizaciĂłn consiste en encontrar el valor de una variable que
maximiza o minimiza una cantidad de interĂ©s (la \textbf{funciĂłn
objetivo}), sujeto a ciertas restricciones geomĂ©tricas o fĂ­sicas.

\begin{pasos}
\begin{enumerate}
    \item \textbf{Diagrama y variables.} Asignar sĂ­mbolos a todas las
    cantidades relevantes.
    \item \textbf{FunciĂłn objetivo.} Escribir la fĂłrmula de la cantidad
    $Q$ que se desea optimizar.
    \item \textbf{ReducciĂłn a una variable.} Usar las restricciones para
    expresar $Q$ en funciĂłn de una sola variable y determinar su dominio.
    \item \textbf{Localizar puntos crĂ­ticos.} Resolver $\frac{dQ}{dx}=0$
    y verificar singularidades o extremos del dominio.
    \item \textbf{Clasificar el extremo.} Usar el criterio de la primera
    o segunda derivada para confirmar que se trata de un mĂĄximo o mĂ­nimo.
\end{enumerate}
\end{pasos}

\subsection{Ejemplos}

\begin{example}
De una hoja rectangular de $24\times 9$ cm se recortan cuadrados iguales
en las esquinas y se doblan los bordes para formar una caja abierta.
ÂżQuĂ© tamaĂ±o deben tener los cuadrados para maximizar el volumen?
\end{example}

\begin{myproof}
Sea $x$ el lado del cuadrado recortado. La caja resultante tiene
dimensiones $(24-2x)\times(9-2x)\times x$, con $0<x<\frac{9}{2}$.
FunciĂłn objetivo:
\[
V(x) = x(24-2x)(9-2x) = x(216-48x-18x+4x^2) = 4x^3-66x^2+216x.
\]
\[
V'(x) = 12x^2-132x+216 = 12(x^2-11x+18) = 12(x-2)(x-9).
\]
En $(0,\frac{9}{2})$, el Ășnico punto crĂ­tico es $x=2$.

$V''(x)=24x-132$; $V''(2)=48-132=-84<0$: mĂĄximo local. Verificando
en los extremos: $V(0)=V(4.5)=0$. Concluimos que el volumen mĂĄximo
se obtiene con $x=2$ cm, y vale $V(2)=2(20)(5)=200$ cm$^3$. $\square$
\end{myproof}

\begin{example}
Se desea construir una lata cilĂ­ndrica (sin tapa) de volumen $V_0$.
Encontrar las dimensiones que minimizan el ĂĄrea de material.
\end{example}

\begin{myproof}
Variables: radio $r$, altura $h$. RestricciĂłn: $\pi r^2 h = V_0$,
de donde $h=\frac{V_0}{\pi r^2}$.

FunciĂłn objetivo (ĂĄrea lateral + base):
\[
A = \pi r^2 + 2\pi r h = \pi r^2 + \frac{2V_0}{r}.
\]
\[
A'(r) = 2\pi r - \frac{2V_0}{r^2} = 0
\implies r^3 = \frac{V_0}{\pi}
\implies r = \left(\frac{V_0}{\pi}\right)^{1/3}.
\]
Con ese valor, $h = \frac{V_0}{\pi r^2} = r$, es decir, la altura
iguala el radio. $A''(r)=2\pi+\frac{4V_0}{r^3}>0$: mĂ­nimo. $\square$
\end{myproof}

\begin{example}
Una tuberĂ­a debe ir desde el punto $A$ en una orilla de un rĂ­o de
$3$ km de ancho hasta el punto $B$, situado $8$ km aguas abajo en la
orilla opuesta. Tender tuberĂ­a por el rĂ­o cuesta \$200/km y por tierra
\$100/km. ÂżDĂłnde debe cruzar el rĂ­o para minimizar el costo total?
\end{example}

\begin{myproof}
Sea $x$ la distancia (en km) medida desde el punto directamente opuesto
a $A$ hasta el punto de cruce $C$ en la orilla de $B$, con
$0\leq x\leq 8$.

Longitud por el rĂ­o: $\sqrt{9+x^2}$ km. Longitud por tierra: $(8-x)$ km.
\[
C(x) = 200\sqrt{9+x^2}+100(8-x).
\]
\[
C'(x) = \frac{200x}{\sqrt{9+x^2}}-100 = 0
\implies \frac{2x}{\sqrt{9+x^2}} = 1
\implies 4x^2 = 9+x^2
\implies x = \sqrt{3} \approx 1.73 \text{ km}.
\]
$C''(x)=\frac{200\cdot 9}{(9+x^2)^{3/2}}>0$: mĂ­nimo.
El costo mĂ­nimo es $C(\sqrt{3})=200\sqrt{12}+100(8-\sqrt{3})
\approx 1292.8$ USD. $\square$
\end{myproof}

\section{El Teorema del Valor Medio}
%%==========================================================

\begin{theorem}[Rolle]
Si $f$ es continua en $[a,b]$, derivable en $(a,b)$ y $f(a)=f(b)$,
entonces existe al menos un $c\in(a,b)$ tal que $f'(c)=0$.
\end{theorem}

\begin{theorem}[Valor Medio --- Lagrange]
Si $f$ es continua en $[a,b]$ y derivable en $(a,b)$, existe al menos
un $c\in(a,b)$ tal que
\[
f'(c) = \frac{f(b)-f(a)}{b-a}.
\]
\end{theorem}

GeomĂ©tricamente, el Teorema del Valor Medio (TVM) garantiza que existe
un punto en la curva donde la recta tangente es \textbf{paralela} a la
recta secante que une los extremos del intervalo. FĂ­sicamente: si un
automĂłvil recorre $120$ km en $2$ horas (velocidad promedio $60$ km/h),
entonces en algĂșn instante exacto su velocidad instantĂĄnea fue
exactamente $60$ km/h.

\begin{myproof}[DemostraciĂłn del TVM a partir de Rolle]
Definir la funciĂłn auxiliar
\[
h(x) = f(x) - f(a) - \frac{f(b)-f(a)}{b-a}(x-a).
\]
$h$ es continua en $[a,b]$ y derivable en $(a,b)$. AdemĂĄs
$h(a)=h(b)=0$. Por el Teorema de Rolle, existe $c\in(a,b)$ con
$h'(c)=0$, es decir:
\[
f'(c) - \frac{f(b)-f(a)}{b-a} = 0
\implies f'(c) = \frac{f(b)-f(a)}{b-a}. \qquad\square
\]
\end{myproof}

\begin{coro}
Si $F'(x)=G'(x)$ para todo $x\in(a,b)$, entonces $F(x)=G(x)+C$
para alguna constante $C$.
\end{coro}

\begin{rem}
Este corolario es el fundamento de la unicidad de la antiderivada:
dos funciones con la misma derivada solo pueden diferir en una
constante aditiva. Es la base teĂłrica del ``$+C$'' en la integraciĂłn
indefinida.
\end{rem}

\begin{example}

\end{example}

\begin{myproof}
$f(0)=0$, $f(2)=6$. Pendiente de la secante: $\frac{6-0}{2-0}=3$.
$f'(x)=3x^2-1=3 \Rightarrow x^2=\frac{4}{3}
\Rightarrow c=\frac{2}{\sqrt{3}}\approx 1.155\in(0,2)$. $\square$
\end{myproof}

\section{Regla de L'HĂŽpital}
%%==========================================================

\begin{theorem}[L'HĂŽpital]
SupĂłngase que $\lim_{x\to a}f(x)=0$ y $\lim_{x\to a}g(x)=0$ (forma
$\frac{0}{0}$), o que ambos tienden a $\pm\infty$ (forma
$\frac{\infty}{\infty}$). Si $f$ y $g$ son derivables cerca de $a$,
$g'(x)\neq 0$ allĂ­, y el lĂ­mite $\lim_{x\to a}\frac{f'(x)}{g'(x)}$
existe (o es $\pm\infty$), entonces
\[
\lim_{x\to a}\frac{f(x)}{g(x)} = \lim_{x\to a}\frac{f'(x)}{g'(x)}.
\]
El resultado tambiĂ©n vale para $a=\pm\infty$ y para lĂ­mites laterales.
\end{theorem}

\begin{rem}
La regla de L'HĂŽpital se aplica derivando \textbf{numerador y denominador
por separado}, no usando la regla del cociente. Dos errores frecuentes:
(1) aplicarla cuando el lĂ­mite \emph{no} es una forma indeterminada,
obteniendo un resultado incorrecto; (2) aplicarla indefinidamente sin
verificar que el nuevo lĂ­mite sea mĂĄs simple. Si despuĂ©s de una
aplicaciĂłn el lĂ­mite sigue siendo indeterminado, puede aplicarse de
nuevo; si oscila, la regla no es Ăștil y se debe buscar otro mĂ©todo.
\end{rem}

\subsection{ReducciĂłn de otras formas indeterminadas}

\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{l l l}
\toprule
\textbf{Forma} & \textbf{TransformaciĂłn} & \textbf{Reduce a} \\
\midrule
$0\cdot\infty$ & $f\cdot g = \dfrac{f}{1/g}$ o $\dfrac{g}{1/f}$
    & $0/0$ o $\infty/\infty$ \\
$\infty-\infty$ & denominador comĂșn o racionalizaciĂłn
    & $0/0$ \\
$0^0,\;1^\infty,\;\infty^0$ & $y=f^g \Rightarrow \ln y = g\ln f$,
    luego hallar $\lim\ln y = L$
    & $0/0$ o $\infty/\infty$; resultado: $e^L$ \\
\bottomrule
\end{tabular}
\end{center}

\subsection{Ejemplos}

\begin{example}
Calcular $\displaystyle\lim_{x\to 0}\frac{e^x-1-x}{x^2}$.
\end{example}

\begin{myproof}
Forma $\frac{0}{0}$. Primera aplicaciĂłn:
$\lim_{x\to 0}\frac{e^x-1}{2x}$, sigue $\frac{0}{0}$.
Segunda aplicaciĂłn:
$\lim_{x\to 0}\frac{e^x}{2} = \frac{1}{2}$. $\square$
\end{myproof}

\begin{example}
Calcular $\displaystyle\lim_{x\to\infty}x\,e^{-x}$.
\end{example}

\begin{myproof}
Forma $\infty\cdot 0$. Reescribimos: $\lim_{x\to\infty}\frac{x}{e^x}$,
forma $\frac{\infty}{\infty}$. L'HĂŽpital: $\lim_{x\to\infty}\frac{1}{e^x}=0$.
$\square$
\end{myproof}

\begin{example}
Calcular $\displaystyle\lim_{x\to 0^+}x^x$.
\end{example}

\begin{myproof}
Forma $0^0$. Sea $y=x^x$; $\ln y = x\ln x$.
$\lim_{x\to 0^+}x\ln x = \lim_{x\to 0^+}\frac{\ln x}{1/x}$,
forma $\frac{-\infty}{\infty}$. L'HĂŽpital:
$\lim_{x\to 0^+}\frac{1/x}{-1/x^2}=\lim_{x\to 0^+}(-x)=0$.
Por tanto $\lim\ln y=0$, asĂ­ que $\lim x^x = e^0=1$. $\square$
\end{myproof}

\begin{example}
Calcular $\displaystyle\lim_{x\to\infty}\left(1+\frac{a}{x}\right)^x$.
\end{example}

\begin{myproof}
Forma $1^\infty$. $\ln y = x\ln\!\left(1+\frac{a}{x}\right)$.
\[
\lim_{x\to\infty} x\ln\!\left(1+\frac{a}{x}\right)
= \lim_{x\to\infty}\frac{\ln(1+a/x)}{1/x}
\;\overset{\text{L'H}}{=}\;
\lim_{x\to\infty}\frac{\frac{-a/x^2}{1+a/x}}{-1/x^2}
= \lim_{x\to\infty}\frac{a}{1+a/x} = a.
\]
Por tanto $\lim y = e^a$. Para $a=1$ se recupera el lĂ­mite notable
$\lim_{x\to\infty}(1+1/x)^x=e$. $\square$
\end{myproof}

\subsection{BisecciĂłn}

El mĂ©todo de bisecciĂłn es una aplicaciĂłn directa del Teorema de Bolzano.
Si $f$ es continua en $[a,b]$ y $f(a)\cdot f(b)<0$, se evalĂșa el punto
medio $m=\frac{a+b}{2}$. SegĂșn el signo de $f(m)$, la raĂ­z queda en
$[a,m]$ o en $[m,b]$; se actualiza el intervalo y se repite. Tras $n$
iteraciones, el error de aproximaciĂłn es menor que $\frac{b-a}{2^n}$.

\begin{rem}
La bisecciĂłn es el mĂ©todo mĂĄs \textbf{robusto}: converge siempre que
se cumplan las hipĂłtesis, independientemente de la forma de $f$.
Su desventaja es la velocidad: la convergencia es \textbf{lineal}
(el nĂșmero de cifras decimales correctas crece linealmente con el
nĂșmero de iteraciones). Para obtener $k$ cifras adicionales se
necesitan aproximadamente $3.32\,k$ iteraciones.
\end{rem}

\subsection{MĂ©todo de Newton-Raphson}

Dada una aproximaciĂłn inicial $x_0$, el mĂ©todo de Newton construye la
recta tangente a la curva en $(x_n, f(x_n))$ y toma su intersecciĂłn
con el eje $x$ como nueva aproximaciĂłn:
\[
x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)}.
\]

\begin{myproof}[DerivaciĂłn geomĂ©trica]
La recta tangente en $(x_n, f(x_n))$ tiene ecuaciĂłn
$y - f(x_n) = f'(x_n)(x-x_n)$. Igualando $y=0$:
$x = x_n - \frac{f(x_n)}{f'(x_n)}$, que es $x_{n+1}$. $\square$
\end{myproof}

La convergencia de Newton-Raphson es \textbf{cuadrĂĄtica}: el nĂșmero de
cifras decimales correctas se \emph{duplica} en cada iteraciĂłn cuando
se estĂĄ suficientemente cerca de la raĂ­z. Sin embargo, el mĂ©todo puede
fallar si $f'(x_n)=0$ (divisiĂłn por cero), si la aproximaciĂłn inicial
estĂĄ lejos de la raĂ­z, o si $f$ tiene oscilaciones que alejan las
iteraciones de la soluciĂłn.

\begin{example}
Aproximar $\sqrt{2}$ usando Newton-Raphson con $x_0=1$.
\end{example}

\begin{myproof}
$f(x)=x^2-2$, $f'(x)=2x$. La iteraciĂłn es
$x_{n+1}=x_n-\frac{x_n^2-2}{2x_n}=\frac{x_n+2/x_n}{2}$.
\begin{center}
\begin{tabular}{c c}
\toprule
$n$ & $x_n$ \\
\midrule
$0$ & $1$ \\
$1$ & $1.5$ \\
$2$ & $1.41\overline{6}$ \\
$3$ & $1.41421356\ldots$ \\
\bottomrule
\end{tabular}
\end{center}
Tres iteraciones dan $8$ cifras correctas. $\square$
\end{myproof}

\subsection{MĂ©todo de punto fijo}

Se reescribe $f(x)=0$ como $x=g(x)$ y se itera $x_{n+1}=g(x_n)$.

\begin{theorem}[Convergencia del punto fijo]
Si $|g'(x)|<1$ en un intervalo que contiene a la raĂ­z, las iteraciones
convergen a ella desde cualquier punto inicial en ese intervalo.
\end{theorem}

\begin{rem}
La condiciĂłn $|g'(x)|<1$ se denomina \textbf{condiciĂłn de contracciĂłn}:
la funciĂłn $g$ ``acerca'' los puntos entre sĂ­ en cada iteraciĂłn. Si
$|g'(x)|>1$, las iteraciones se alejan de la raĂ­z. La elecciĂłn de la
reescritura $x=g(x)$ no es Ășnica, y una misma ecuaciĂłn puede tener
reescrituras convergentes y divergentes; elegir la adecuada es parte
del arte del mĂ©todo.
\end{rem}

& \textbf{Requisitos} \\
\midrule

Newton-Raphson & CuadrĂĄtica & Media & $f$ derivable, buen $x_0$ \\

\bottomrule
\end{tabular}
\end{center}
