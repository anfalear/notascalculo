\chapter{Aplicaciones de la integral}

La integral definida, mĆ”s allĆ” de su interpretaciĆ³n como Ć”rea bajo una curva,
es una herramienta fundamental para modelar y resolver problemas geomĆ©tricos,
fĆ­sicos y de ingenierĆ­a. En este capĆ­tulo desarrollaremos las aplicaciones
clĆ”sicas: cĆ”lculo de Ć”reas de regiones planas, volĆŗmenes de sĆ³lidos de
revoluciĆ³n, longitud de arco, Ć”reas de superficies de revoluciĆ³n y
aplicaciones fĆ­sicas como trabajo, centro de masa y momento de inercia. El
hilo conductor de todo el capĆ­tulo es el mĆ©todo de
\textbf{rebanar, aproximar e integrar}: identificar la cantidad diferencial
$dQ$ apropiada, sumar infinitesimalmente las contribuciones y reconocer la
suma lĆ­mite como una integral definida.

En general, un protocolo prĆ”ctico puede resumirse asĆ­ para cada aplicaciĆ³n:

\begin{rem}
\begin{description}
  \item[Ćreas.] Grafique primero. Si las curvas se expresan mejor como
    $x=g(y)$, integre respecto a $y$. Cuando haya varios cruces, sume
    integrales en subintervalos.

  \item[VolĆŗmenes.]
    \begin{itemize}
      \item \emph{Discos/arandelas:} el rectĆ”ngulo representativo es
        \textbf{perpendicular} al eje de rotaciĆ³n.
      \item \emph{Cascarones:} el rectĆ”ngulo representativo es
        \textbf{paralelo} al eje de rotaciĆ³n.
      \item Para ejes desplazados, ajuste el radio sumando o restando el
        desplazamiento.
    \end{itemize}

  \item[Longitud de arco.] Verifique que $f'$ sea continua; si la integral
    resultante no es elemental, use integraciĆ³n numĆ©rica (Simpson, Trapecios
    o Riemann).

  \item[Superficies.] Recuerde el factor $2\pi\times\text{radio}\times ds$;
    el radio es la distancia del punto al eje de rotaciĆ³n.

  \item[Trabajo.] Identifique $F(x)$ e integre. Para bombeo, exprese la
    distancia de elevaciĆ³n en funciĆ³n de la variable de integraciĆ³n.

  \item[Centro de masa.] Para regiones entre dos curvas:
    $M_y=\rho\int_a^b x[f(x)-g(x)]\,dx$ y
    $M_x=\frac{\rho}{2}\int_a^b([f(x)]^2-[g(x)]^2)\,dx$.

  \item[Fuerza hidrostĆ”tica.] Use $F=\int\rho g h\,w(h)\,dh$; exprese la
    anchura $w(h)$ en funciĆ³n de la profundidad $h$.
\end{description}
\end{rem}

\begin{theorem}[Principio unificador de las aplicaciones de la integral]
Cualquier cantidad acumulativa $Q$ que satisfaga:
\begin{enumerate}
  \item Puede aproximarse en subintervalos: $Q\approx\sum g(x_i^*)\Delta x$.
  \item La aproximaciĆ³n converge cuando $\Delta x\to 0$.
\end{enumerate}
puede calcularse exactamente como $Q=\int_a^b g(x)\,dx$. Este principio
unifica todas las aplicaciones presentadas en el capĆ­tulo: el Ć”rea, el
volumen, la longitud de arco, el trabajo, los momentos y la fuerza
hidrostĆ”tica son, en esencia, lĆ­mites de sumas de Riemann con distintos
integrandos $g(x)$.
\end{theorem}

\begin{rem}
La potencia del cĆ”lculo integral radica en transformar problemas
geomĆ©tricos y fĆ­sicos complejos en operaciones algebraicas sistemĆ”ticas.
Dominar estas aplicaciones requiere prĆ”ctica en tres pasos fundamentales:
\begin{enumerate}
\item Identificar el elemento diferencial $dQ$ adecuado para el problema,
\item Expresar $dQ$ en tĆ©rminos de una sola variable de integraciĆ³n, y
\item Determinar los lĆ­mites de integraciĆ³n a partir de la geometrĆ­a del
problema.
\end{enumerate}
\end{rem}

\begin{definition}[Ćrea bajo una curva]
Sea $f:[a,b]\to\mathbb{R}$ una funciĆ³n continua y no negativa. El
\textbf{Ć”rea} de la regiĆ³n acotada por la grĆ”fica de $f$, el eje $x$ y las
rectas verticales $x=a$, $x=b$ se define como
\[
  A = \int_a^b f(x)\,dx.
\]
Si $f(x)\leq 0$ en $[a,b]$, el Ć”rea geomĆ©trica es
$A = -\int_a^b f(x)\,dx = \int_a^b |f(x)|\,dx$.
\end{definition}

\begin{rem}
La integral definida calcula el \textbf{Ć”rea algebraica} (con signo),
mientras que el Ć”rea geomĆ©trica siempre es no negativa. Para obtener el
Ć”rea real entre una curva y el eje $x$ cuando $f$ cambia de signo, debe
dividirse el intervalo en partes donde $f$ no cambie de signo e integrar
el valor absoluto en cada parte:
\[
  A = \int_a^b |f(x)|\,dx
      = \int_{\{f\ge 0\}} f(x)\,dx \;-\; \int_{\{f\le 0\}} f(x)\,dx.
\]
\end{rem}

\begin{theorem}[Ćrea entre dos curvas]
Sean $f$ y $g$ funciones continuas en $[a,b]$ tales que $f(x)\geq g(x)$
para todo $x\in[a,b]$. El Ć”rea de la regiĆ³n acotada por ambas grĆ”ficas y
las rectas $x=a$, $x=b$ es
\[
  A = \int_a^b \bigl[f(x)-g(x)\bigr]\,dx.
\]
\end{theorem}

\begin{proof}
Dividimos $[a,b]$ en $n$ subintervalos de ancho $\Delta x=(b-a)/n$.
En cada subintervalo $[x_{i-1},x_i]$ se toma $x_i^*$ y se aproxima la
porciĆ³n de la regiĆ³n por un rectĆ”ngulo de altura $f(x_i^*)-g(x_i^*)$ y
base $\Delta x$. La suma de Riemann asociada es
\[
  S_n = \sum_{i=1}^n\bigl[f(x_i^*)-g(x_i^*)\bigr]\Delta x.
\]
Tomando el lĆ­mite cuando $n\to\infty$ y usando la continuidad de $f-g$:
\[
  A = \lim_{n\to\infty} S_n
    = \int_a^b\bigl[f(x)-g(x)\bigr]\,dx.\qedhere
\]
\end{proof}

\section{VolĆŗmenes de sĆ³lidos de revoluciĆ³n}
% ============================================================

Un \textbf{sĆ³lido de revoluciĆ³n} se genera al rotar una regiĆ³n plana
alrededor de una recta llamada \emph{eje de revoluciĆ³n}. Presentamos tres
mĆ©todos fundamentales, cada uno adecuado para distintas configuraciones.

% -----------------------------------------------------------
\subsection{MĆ©todo de los discos}
% -----------------------------------------------------------

\begin{theorem}[Volumen por discos respecto al eje $x$]
Sea $R$ la regiĆ³n acotada por $y=f(x)\geq 0$, $x=a$, $x=b$ y el eje $x$.
Al rotar $R$ alrededor del eje $x$, el volumen del sĆ³lido generado es
\[
  V = \pi\int_a^b\bigl[f(x)\bigr]^2\,dx.
\]
\end{theorem}

\begin{proof}
Dividimos $[a,b]$ en $n$ subintervalos de ancho $\Delta x$. La porciĆ³n del
sĆ³lido sobre $[x_{i-1},x_i]$ se aproxima por un disco circular de radio
$f(x_i^*)$ y grosor $\Delta x$, cuyo volumen es $\pi[f(x_i^*)]^2\Delta x$.
Sumando y tomando lĆ­mite:
\[
  V = \lim_{n\to\infty}\sum_{i=1}^n\pi[f(x_i^*)]^2\Delta x
    = \pi\int_a^b[f(x)]^2\,dx.\qedhere
\]
\end{proof}

\begin{example}
Demuestre que el volumen de un cono de radio $R$ y altura $h$ es
$V=\frac{1}{3}\pi R^2 h$.

\begin{myproof}
\textbf{Paso 1: Generatriz y regiĆ³n.} La recta que genera el cono es
$y=\frac{R}{h}x$, $x\in[0,h]$.

\textbf{Paso 2: Elemento diferencial.} Disco de radio $f(x)$ y grosor $dx$:
$dV = \pi [f(x)]^2 dx$.

\textbf{Paso 3: LĆ­mites.} $x$ va de $0$ a $h$.

\textbf{Paso 4: EvaluaciĆ³n.}
\[
  V = \pi\int_0^h\!\left(\frac{R}{h}x\right)^{\!2}dx
    = \frac{\pi R^2}{h^2}\int_0^h x^2\,dx
    = \frac{\pi R^2}{h^2}\cdot\frac{h^3}{3}
    = \boxed{\frac{1}{3}\pi R^2 h}.
\]
\end{myproof}
\end{example}

\begin{example}
Demuestre que el volumen de una esfera de radio $R$ es
$V = \frac{4}{3}\pi R^3$.

\begin{myproof}
\textbf{Paso 1: Generatriz y regiĆ³n.} Rotamos la semicircunferencia
$y=\sqrt{R^2-x^2}$, $x\in[-R,R]$, alrededor del eje $x$.

\textbf{Paso 2: Elemento diferencial.}
$dV = \pi y^2 dx = \pi(R^2-x^2)dx$.

\textbf{Paso 3: LĆ­mites.} $x\in[-R,R]$. Por simetrĆ­a par,
$V = 2\pi\int_0^R (R^2-x^2)dx$.

\textbf{Paso 4: EvaluaciĆ³n.}
\[
  V = 2\pi\left[R^2 x-\frac{x^3}{3}\right]_0^R
    = 2\pi\left(R^3-\frac{R^3}{3}\right)
    = 2\pi\cdot\frac{2R^3}{3}
    = \boxed{\dfrac{4}{3}\pi R^3}.\qedhere
\]
\end{myproof}
\end{example}

\section{Longitud de arco}
% ============================================================

\begin{definition}[Longitud de una curva suave]
Sea $f$ una funciĆ³n con derivada continua en $[a,b]$. La
\textbf{longitud de arco} de la grĆ”fica de $f$ desde $x=a$ hasta $x=b$ es
\[
  s = \int_a^b\sqrt{1+\bigl[f'(x)\bigr]^2}\,dx.
\]
\end{definition}

\begin{proof}[Esquema de demostraciĆ³n]
Dividimos $[a,b]$ en $n$ subintervalos. La longitud del segmento que une
$(x_{i-1},f(x_{i-1}))$ con $(x_i,f(x_i))$ es, por el teorema de PitĆ”goras,
\[
  \Delta s_i = \sqrt{(\Delta x_i)^2+(\Delta y_i)^2}
             = \sqrt{1+\!\left(\frac{\Delta y_i}{\Delta x_i}\right)^{\!2}}
               \,\Delta x_i.
\]
Por el Teorema del Valor Medio, $\dfrac{\Delta y_i}{\Delta x_i}=f'(c_i)$
para algĆŗn $c_i\in(x_{i-1},x_i)$. Sumando y tomando lĆ­mite:
\[
  s = \lim_{n\to\infty}\sum_{i=1}^n\sqrt{1+[f'(c_i)]^2}\,\Delta x_i
    = \int_a^b\sqrt{1+[f'(x)]^2}\,dx.\qedhere
\]
\end{proof}

\begin{rem}
El elemento diferencial de longitud de arco es
$ds=\sqrt{1+[f'(x)]^2}\,dx = \sqrt{dx^2+dy^2}$, que corresponde
geomĆ©tricamente a la hipotenusa del triĆ”ngulo diferencial de catetos $dx$
y $dy=f'(x)\,dx$. Esta interpretaciĆ³n facilita recordar y derivar la
fĆ³rmula en distintos contextos (paramĆ©trico, polar, etc.).
\end{rem}

\begin{example}[Longitud de una parĆ”bola]
Calcule la longitud de arco de $y=x^2$ en $[0,1]$.

\begin{myproof}
\textbf{Paso 1: Derivada e integrando.}
$f'(x)=2x \Rightarrow ds = \sqrt{1+4x^2}\,dx$.

\textbf{Paso 2: LĆ­mites.} $x\in[0,1]$.

\textbf{Paso 3: EvaluaciĆ³n.}
Con la sustituciĆ³n trigonomĆ©trica $2x=\tan\theta$,
$dx=\frac{1}{2}\sec^2\theta\,d\theta$:
\begin{align*}
  s &= \frac{1}{2}\int_0^{\arctan 2}\sec^3\theta\,d\theta
     = \frac{1}{4}\Bigl[\sec\theta\tan\theta
       +\ln|\sec\theta+\tan\theta|\Bigr]_0^{\arctan 2} \\
    &= \frac{1}{4}\bigl(\sqrt{5}\cdot 2+\ln(2+\sqrt{5})\bigr)
     = \boxed{\dfrac{\sqrt{5}}{2}+\dfrac{1}{4}\ln(2+\sqrt{5})\approx 1.479}.
\end{align*}
\end{myproof}
\end{example}

\begin{example}[Longitud del arco de la catenaria]
Calcule la longitud de arco de $y=\cosh x = \dfrac{e^x+e^{-x}}{2}$ en
$[0,\ln 2]$.

\begin{myproof}
\textbf{Paso 1: Derivada e integrando.}
$f'(x)=\sinh x$. Usando $\cosh^2 x-\sinh^2 x=1$:
$ds = \sqrt{1+\sinh^2 x}\,dx = \cosh x\,dx$.

\textbf{Paso 2: LĆ­mites.} $x\in[0,\ln 2]$.

\textbf{Paso 3: EvaluaciĆ³n.}
\[
  s = \int_0^{\ln 2}\cosh x\,dx
    = \bigl[\sinh x\bigr]_0^{\ln 2}
    = \sinh(\ln 2)
    = \frac{e^{\ln 2} - e^{-\ln 2}}{2}
    = \frac{2 - \frac{1}{2}}{2}
    = \boxed{\dfrac{3}{4}}.
\]
\end{myproof}
\end{example}

\section{Ćreas de superficies de revoluciĆ³n}
% ============================================================

\begin{definition}[Ćrea superficial de revoluciĆ³n --- eje $x$]
Sea $f$ una funciĆ³n no negativa con derivada continua en $[a,b]$. El Ć”rea
de la superficie generada al rotar la grĆ”fica de $f$ alrededor del eje $x$
es
\[
  S = 2\pi\int_a^b f(x)\sqrt{1+[f'(x)]^2}\,dx.
\]
\end{definition}

\begin{proof}[Idea de demostraciĆ³n]
Un pequeĆ±o arco de longitud $ds=\sqrt{1+[f'(x)]^2}\,dx$, al rotar alrededor
del eje $x$, genera una banda (frustum de cono) cuya Ć”rea aproximada es
\[
  dS = 2\pi f(x)\,ds = 2\pi f(x)\sqrt{1+[f'(x)]^2}\,dx.
\]
Integrando: $S = 2\pi\int_a^b f(x)\sqrt{1+[f'(x)]^2}\,dx$. \qedhere
\end{proof}

\begin{rem}
El factor $2\pi f(x)$ es la circunferencia del cĆ­rculo descrito por el
punto $(x,f(x))$ al rotar. El factor $ds$ es el elemento de longitud de
arco, que ``estira'' la banda segĆŗn la inclinaciĆ³n de la curva. Si se
tomara $ds=dx$ en lugar de $ds=\sqrt{1+[f'(x)]^2}\,dx$ se obtendrĆ­a la

\begin{definition}[Momentos y centro de masa de una lĆ”mina plana]

  \bar{x} = \frac{M_y}{m}, \qquad \bar{y} = \frac{M_x}{m}.

\end{theorem}

\begin{rem}

\begin{definition}[Momento de inercia de una lĆ”mina]

\end{definition}

\begin{example}[Momento de inercia de una varilla]

\]
\end{myproof}
\end{example}

\begin{example}[Momento de inercia respecto al centro]

$I=I_{\text{cm}}+M d^2$, donde $d$ es la distancia entre los ejes.
\end{myproof}
\end{example}
