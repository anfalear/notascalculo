\chapter{Funciones}\label{funciones}

La funciУГn es, sin duda, el concepto mУЁs omnipresente en las matemУЁticas: toda la arquitectura del cУЁlculo diferencial e integral descansa sobre УЉl. Antes de precisar la definiciУГn formal, conviene cultivar la intuiciУГn correcta. Imagine una \emph{mУЁquina procesadora de nУКmeros}: se introduce un valor $x$ en la ranura de entrada, la mУЁquina ejecuta una operaciУГn interna y devuelve un УКnico resultado $f(x)$ por la ranura de salida. Esta metУЁfora captura la esencia de la dependencia funcional: el valor de salida \emph{depende} del valor de entrada, y a cada entrada le corresponde exactamente una salida.

En la prУЁctica, esta dependencia aparece en todas partes: el УЁrea de un cУ­rculo depende de su radio, el costo de un envУ­o depende de su peso, la temperatura de un cuerpo depende del tiempo transcurrido. En este capУ­tulo desarrollaremos con rigor y detalle este concepto fundamental, explorando sus propiedades algebraicas y su comportamiento grУЁfico.


\section{Propiedades bУЁsicas}

\begin{definition}[FunciУГn]
Una \textbf{funciУГn} $f$ es una regla de correspondencia que asigna a cada elemento $x$ de un conjunto denominado \textbf{dominio} $D_f$, un \emph{УКnico} valor $f(x)$ en otro conjunto llamado \textbf{codominio}. 

El conjunto de todos los valores efectivamente alcanzados,
\[
  R_f = \{\, f(x) : x \in D_f \,\},
\]
se denomina \textbf{rango} o \textbf{imagen} de $f$, y satisface $R_f \subseteq \text{codominio}$. Escribimos $f: D_f \to \mathbb{R}$ para indicar que $f$ va del dominio al conjunto de los nУКmeros reales. Es convenciУГn llamar a $x$ la \textbf{variable independiente} y a $y = f(x)$ la \textbf{variable dependiente}.
\end{definition}

\begin{rem}[La regla del УКnico valor de salida]
La unicidad del valor de salida es la condiciУГn definitoria de una funciУГn. Si a un valor $x$ le correspondieran dos salidas distintas $y_1 \neq y_2$, la regla no serУ­a una funciУГn. Esta unicidad se verifica grУЁficamente mediante la \emph{prueba de la recta vertical}.
\end{rem}

\subsection{Dominio natural de una funciУГn}
% -----------------------------------------------------------

Cuando una funciУГn se presenta mediante una fУГrmula algebraica sin restricciones explУ­citas sobre su dominio, el \textbf{dominio natural} (o dominio implУ­cito) es el conjunto mУЁs amplio de nУКmeros reales para los cuales la expresiУГn estУЁ definida. Las dos fuentes principales de restricciУГn son:

\begin{itemize}
  \item \textbf{Denominadores:} se excluyen los valores que anulan el denominador.
  \item \textbf{Radicales de У­ndice par:} se requiere que el radicando sea no negativo.
\end{itemize}



\begin{prob}\label{prob:dominio-basico}
Calcule el dominio natural de cada funciУГn y exprУЉselo en notaciУГn de intervalos.
\begin{enumerate}
  \item $f(x) = x^2 - 8x + 5$
  \item $f(x) = \sqrt{1 - x^2}$
  \item $f(x) = \sqrt{3+x} + \sqrt[4]{7-x}$
  \item $f(x) = \dfrac{x+5}{x-3}$
\end{enumerate}

\begin{myproof}
\textbf{1.} \emph{AnУЁlisis:} los polinomios no tienen restricciones.

\emph{FundamentaciУГn:} la funciУГn es un polinomio de grado $2$; los polinomios estУЁn definidos para todo $x \in \mathbb{R}$.

\emph{ConclusiУГn:}
\[
  \boxed{D_f = \mathbb{R} = (-\infty,+\infty).}
\]

\medskip
\textbf{2.} \emph{AnУЁlisis:} la raУ­z cuadrada requiere radicando no negativo, lo que acotarУЁ $x$ a un intervalo finito.

\emph{FundamentaciУГn:} necesitamos $1 - x^2 \geq 0$. Factorizando:
\[
  1 - x^2 = (1-x)(1+x) \geq 0.
\]

\emph{Procedimiento:} construimos la tabla de signos:

\begin{table}[H]
\centering
\begin{tabular}{|c|c|c|c|}
\hline
Intervalo & $1+x$ & $1-x$ & $(1-x)(1+x)$ \\
\hline
$(-\infty,-1)$ & $-$ & $+$ & $-$ \\
\hline
$[-1,\,1]$      & $+$ & $+$ & $+$ \\
\hline
$(1,+\infty)$   & $+$ & $-$ & $-$ \\
\hline
\end{tabular}
\caption{Tabla de signos para el У­tem 2 del Problema \ref{prob:dominio-basico}.}
\end{table}

\emph{ConclusiУГn:}
\[
  \boxed{D_f = [-1,\,1].}
\]

\medskip
\textbf{3.} \emph{AnУЁlisis:} dos radicales de У­ndice par imponen dos condiciones simultУЁneas.

\emph{Procedimiento:} la raУ­z cuarta tiene У­ndice par, por lo que tambiУЉn requiere radicando no negativo.
\begin{align*}
  \sqrt{3+x}:   &\quad 3 + x \geq 0 \;\Rightarrow\; x \geq -3, \\
  \sqrt[4]{7-x}:&\quad 7 - x \geq 0 \;\Rightarrow\; x \leq 7.
\end{align*}
La suma estУЁ definida cuando \emph{ambas} condiciones se cumplen.

\emph{ConclusiУГn:}
\[
  \boxed{D_f = [-3,\,7].}
\]

\medskip
\textbf{4.} \emph{AnУЁlisis:} el denominador no puede ser cero.

\emph{Procedimiento:} el denominador se anula en $x = 3$.

\emph{ConclusiУГn:}
\[
  \boxed{D_f = \mathbb{R} \setminus \{3\} = (-\infty, 3)\cup(3,+\infty).}
\]
\end{myproof}
\end{prob}

\subsection{Cuatro formas de representar una funciУГn}
% -----------------------------------------------------------

\begin{rem}[Formas de representar una funciУГn]
Existen cuatro formas complementarias de representar una funciУГn:
\begin{enumerate}
  \item \textbf{Verbal:} descripciУГn en palabras de la relaciУГn entre variables.
  \item \textbf{NumУЉrica:} tabla de valores $(x, f(x))$.
  \item \textbf{Algebraica:} fУГrmula explУ­cita $y = f(x)$.
  \item \textbf{Visual:} grУЁfica en el plano $xy$.
\end{enumerate}
Cada representaciУГn revela aspectos distintos del comportamiento de la funciУГn; la habilidad del matemУЁtico consiste en elegir la mУЁs conveniente para cada situaciУГn.
\end{rem}


\begin{example}\label{ex:costo-contenedor}
Un contenedor rectangular sin tapa tiene un volumen de $10\,\mathrm{m}^3$. La longitud de su base es el doble de su ancho. El material para la base cuesta \$10 por metro cuadrado y el material para los cuatro lados cuesta \$6 por metro cuadrado. Exprese el costo de los materiales como funciУГn del ancho de la base, proporcionando las cuatro representaciones.

\begin{myproof}
\textbf{Paso 1 т AnУЁlisis grУЁfico y tabular.} Sea $x > 0$ el ancho de la base (en metros). Entonces la longitud es $2x$ y la altura $h$ se determinarУЁ por la condiciУГn de volumen. Anticipamos que $C(x)$ crecerУЁ cuando $x \to 0^+$ (los lados se vuelven muy altos) y cuando $x \to +\infty$ (la base domina el costo), por lo que tendrУЁ un mУ­nimo.

\textbf{Paso 2 т FundamentaciУГn.} Las УЁreas de las superficies determinan el costo; la condiciУГn de volumen fija $h$ en funciУГn de $x$.

\textbf{Paso 3 т Procedimiento algebraico.} De la condiciУГn de volumen:
\[
  V = x \cdot 2x \cdot h = 2x^2 h = 10 \;\Rightarrow\; h = \frac{5}{x^2}.
\]
Уreas y costos parciales:
\begin{align*}
  A_{\text{base}} &= 2x^2, & C_{\text{base}} &= 10 \cdot 2x^2 = 20x^2,\\
  A_{\text{lados}} &= 2(xh) + 2(2xh) = 6xh = \frac{30}{x}, & C_{\text{lados}} &= 6 \cdot \frac{30}{x} = \frac{180}{x}.
\end{align*}

\textbf{Paso 4 т ConclusiУГn y representaciones.}

\textbf{Algebraica:}
\[
  \boxed{C(x) = 20x^2 + \frac{180}{x},\quad x > 0.}
\]

\textbf{Verbal:} A cada ancho positivo $x$ de la base se asocia el costo total de construir el contenedor, sumando el aporte de la base cuadrada (que crece con $x^2$) y el de los cuatro lados (que decrecen con $x$).

\textbf{NumУЉrica:}
\[
\begin{array}{c|c}
x\,(\mathrm{m}) & C(x)\,(\$) \\\hline
1   & 200 \\
1.5 & 165 \\
2   & 170 \\
2.5 & 197 \\
\end{array}
\]

\textbf{Visual:} vУЉase la figura~\ref{fig:costo_contenedor}.

\subsection{Funciones crecientes, decrecientes, pares e impares}
% -----------------------------------------------------------

\begin{definition}[Funciones monУГtonas]
Sea $f$ una funciУГn definida en un intervalo $I$.
\begin{itemize}
  \item $f$ es \textbf{creciente} en $I$ si $x_1 < x_2 \Rightarrow f(x_1) < f(x_2)$ para todo $x_1, x_2 \in I$.
  \item $f$ es \textbf{decreciente} en $I$ si $x_1 < x_2 \Rightarrow f(x_1) > f(x_2)$ para todo $x_1, x_2 \in I$.
\end{itemize}
\end{definition}

\begin{example}
Determine los intervalos de crecimiento y decrecimiento de $f(x) = x^2$.

\begin{myproof}
\textbf{Paso 1 т AnУЁlisis grУЁfico.} La parУЁbola tiene vУЉrtice en el origen y es simУЉtrica respecto al eje $y$; se anticipa crecimiento en $[0,+\infty)$ y decrecimiento en $(-\infty,0]$.

\textbf{Paso 2 т FundamentaciУГn.} Usaremos la definiciУГn: analizamos el signo de $f(x_2) - f(x_1)$ para $x_1 < x_2$.

\textbf{Paso 3 т Procedimiento.}
\[
  f(x_2) - f(x_1) = x_2^2 - x_1^2 = (x_2 - x_1)(x_2 + x_1).
\]
Como $x_1 < x_2$, se tiene $x_2 - x_1 > 0$. El signo de la diferencia depende de $x_2 + x_1$:
\begin{itemize}
  \item Si $x_1, x_2 \geq 0$: $x_2 + x_1 \geq 0$, luego $f(x_2) - f(x_1) \geq 0$. ($f$ es creciente.)
  \item Si $x_1, x_2 \leq 0$: $x_2 + x_1 \leq 0$, luego $f(x_2) - f(x_1) \leq 0$. ($f$ es decreciente.)
\end{itemize}

\textbf{Paso 4 т ConclusiУГn:}
\[
  \boxed{f \text{ es decreciente en } (-\infty, 0] \text{ y creciente en } [0,+\infty).}
\]
\end{myproof}
\end{example}

El ejemplo anterior se resolverУЁ con mayor facilidad mУЁs adelante, luego de estudiar la derivada. Mientras tanto, observe que hay algunas formas adicionales de representar funciones:

\begin{definition}[Funciones pares e impares]
Sea $f: D_f \to \mathbb{R}$ con $D_f$ simУЉtrico respecto al origen (es decir, $x \in D_f \Rightarrow -x \in D_f$).
\begin{itemize}
  \item $f$ es \textbf{par} si $f(-x) = f(x)$ para todo $x \in D_f$. Su grУЁfica es simУЉtrica respecto al eje $y$.
  \item $f$ es \textbf{impar} si $f(-x) = -f(x)$ para todo $x \in D_f$. Su grУЁfica es simУЉtrica respecto al origen.
\end{itemize}
\end{definition}

\begin{example}[ClasificaciУГn por paridad]
Clasifique $f(x) = x^2$ y $g(x) = x^3$ segУКn paridad.

\begin{myproof}
\textbf{Paso 1 т AnУЁlisis de $f(x)=x^2$:}
\[
  f(-x) = (-x)^2 = x^2 = f(x).
\]
Como $f(-x)=f(x)$, la funciУГn $f$ es \textbf{par}.

\textbf{Paso 2 т AnУЁlisis de $g(x)=x^3$:}
\[
  g(-x) = (-x)^3 = -x^3 = -g(x).
\]
Como $g(-x)=-g(x)$, la funciУГn $g$ es \textbf{impar}.

\textbf{Paso 3 т ConclusiУГn:}
\[
  \boxed{f(x)=x^2 \text{ es par}; \quad g(x)=x^3 \text{ es impar.}}
\]
\end{myproof}
\end{example}

\subsection{FunciУГn potencia}
% -----------------------------------------------------------

\begin{definition}[FunciУГn potencia]
Sea $r \in \mathbb{Q}$, $r = p/q$ en forma reducida con $q \in \mathbb{N}^+$. La \textbf{funciУГn potencia} es $f(x) = x^r = \sqrt[q]{x^p}$. Su dominio depende de la paridad de $q$ y el signo de $r$:
\begin{itemize}
  \item $q$ \textbf{impar}: $D_f = \mathbb{R}$.
  \item $q$ \textbf{par}: requiere $x^p \geq 0$; para $p$ impar, $D_f = [0,+\infty)$.
  \item $r < 0$: se excluye $x = 0$: $D_f = \mathbb{R}\setminus\{0\}$ (o $(0,+\infty)$ si $q$ par).
\end{itemize}
\end{definition}

\begin{rem}
Como casos particulares, para enteros positivos $n \in \mathbb{N}$, $f(x) = x^n$ es un polinomio definido en todo $\mathbb{R}$. Para enteros negativos $n < 0$, $f(x) = x^n = \frac{1}{x^{-n}}$ estУЁ definida en $\mathbb{R} \setminus \{0\}$.

Estas reglas garantizan que $x^r$ produzca valores reales para todo $x$ en su dominio respectivo.
\end{rem}

\subsection{FunciУГn compuesta y funciУГn elemental}
% -----------------------------------------------------------

\begin{definition}[FunciУГn compuesta]
Sean $f: D_f \to \mathbb{R}$ y $g: D_g \to \mathbb{R}$. La \textbf{composiciУГn} $f \circ g$ se define por
\[
  (f \circ g)(x) = f(g(x)),
\]
con dominio $D_{f \circ g} = \{\, x \in D_g : g(x) \in D_f \,\}$.
\end{definition}

\begin{example}
Identifique la composiciУГn en $y = \sqrt{1-x^2}$ y halle su dominio.

\begin{myproof}
\textbf{Paso 1:} La funciУГn exterior es $f(u) = \sqrt{u}$ (requiere $u \geq 0$); la interior es $g(x) = 1-x^2$.

\textbf{Paso 2:} Para $f\circ g$ se necesita $g(x) \in D_f = [0,+\infty)$.

\textbf{Paso 3:} $1-x^2 \geq 0 \Leftrightarrow x^2 \leq 1 \Leftrightarrow -1 \leq x \leq 1$.

\textbf{Paso 4:}
\[
  \boxed{D_{f\circ g} = [-1,\,1].}
\]
\end{myproof}
\end{example}

\begin{rem}[Composiciones mУКltiples]
La funciУГn $h(x) = \ln(\sen(x^2+1))$ es la composiciУГn $h = \ln \circ \sen \circ p$, donde $p(x)=x^2+1$. Para hallar su dominio se imponen las restricciones de adentro hacia afuera: $p(x)=x^2+1 \geq 1 > 0$ (sin restricciУГn); $\sen(p(x))$ debe ser positivo para que $\ln$ estУЉ definido. Luego
\[
  D_h = \{\, x \in \mathbb{R} : \sen(x^2+1) > 0 \,\}.
\]
\end{rem}

\begin{definition}[FunciУГn elemental]
Una funciУГn $f$ es \textbf{elemental} si se obtiene a partir de funciones potencia, exponenciales, logarУ­tmicas, trigonomУЉtricas (y sus inversas) y constantes reales mediante un nУКmero \emph{finito} de sumas, restas, multiplicaciones, divisiones y composiciones.

Las \textbf{funciones algebraicas} son el subconjunto de funciones elementales que solo involucran operaciones algebraicas (sin exponenciales ni trigonomУЉtricas). Se subdividen en:
\begin{itemize}
  \item \textbf{Polinomios:} $f(x) = a_n x^n + \cdots + a_0$, $D_f = \mathbb{R}$.
  \item \textbf{Racionales:} $f(x) = P(x)/Q(x)$, $D_f = \{x : Q(x)\neq 0\}$.
  \item \textbf{Irracionales:} contienen raУ­ces $\sqrt[n]{\cdot}$ con $n \geq 2$.
\end{itemize}
Las funciones no algebraicas (senos, cosenos, exponenciales, logaritmos) se denominan \textbf{trascendentes}.
\end{definition}

\section{Transformaciones de funciones}

\subsection{Dilataciones, compresiones y reflexiones}
% -----------------------------------------------------------

Sea $c > 0$, $c \neq 1$:
\begin{itemize}
  \item $y = c\,f(x)$, $c>1$: \textbf{dilataciУГn vertical} (estira hacia arriba).
  \item $y = \tfrac{1}{c}\,f(x)$, $c>1$: \textbf{compresiУГn vertical}.
  \item $y = f(cx)$, $c>1$: \textbf{compresiУГn horizontal} (acorta el perУ­odo).
  \item $y = f(x/c)$, $c>1$: \textbf{dilataciУГn horizontal} (alarga el perУ­odo).
  \item $y = -f(x)$: \textbf{reflexiУГn respecto al eje $x$}.
  \item $y = f(-x)$: \textbf{reflexiУГn respecto al eje $y$} (no altera paridad).
\end{itemize}

Para $\sen$ y $\cos$, la forma generalizada
\[

tiene los siguientes parУЁmetros geomУЉtricos:
\begin{itemize}
  \item \textbf{Amplitud:} $|A|$ (distancia desde la lУ­nea central hasta los extremos).
  \item \textbf{PerУ­odo fundamental:} $T = 2\pi/|a|$.
  \item \textbf{LУ­nea central (nivel medio):} $y = C$.
  \item \textbf{Desfase horizontal:} $b$ (desplaza el ciclo).
\end{itemize}

\begin{example}[Mareas de la bahУ­a de Fundy]
Las mareas mУЁs altas del mundo ocurren en la bahУ­a de Fundy (CanadУЁ). En
Hopewell Cape, la profundidad mУ­nima es $2.0\,\mathrm{m}$ y la mУЁxima es
$12.0\,\mathrm{m}$; el perУ­odo de oscilaciУГn es de aproximadamente $12$
horas y, el 30 de junio de 2009, la marea alta se produjo a las 6:45\,h.
Determine una funciУГn $D(t)$ ---utilizando la funciУГn coseno--- que modele
la profundidad del agua (en metros) en funciУГn del tiempo $t$ (horas
despuУЉs de la medianoche).

\begin{myproof}
\textbf{Paso 1 т AnУЁlisis grУЁfico y planteamiento.}
La profundidad oscila entre $2$ y $12$ m con perУ­odo de $12$ h: es un
fenУГmeno sinusoidal de la forma $D(t) = C + A\cos[a(t-b)]$. Conviene el
coseno porque alcanza su mУЁximo exactamente cuando el argumento se anula:
basta centrar el desfase $b$ en la hora de la marea alta.

\textbf{Paso 2 т FundamentaciУГn teУГrica.}
En la forma amplitud-fase, la lУ­nea central $C$ es el promedio de los
valores extremos, la amplitud $|A|$ es su semidiferencia y $a$ se obtiene
del perУ­odo mediante $T = 2\pi/|a|$.

\textbf{Paso 3 т Procedimiento tУЉcnico.}
\begin{align*}
  C &= \frac{12.0 + 2.0}{2} = 7.0, &
  A &= \frac{12.0 - 2.0}{2} = 5.0,\\
  a &= \frac{2\pi}{T} = \frac{2\pi}{12} = \frac{\pi}{6}, &
  b &= 6.75 \quad (\text{6:45\,h} = 6\tfrac{3}{4}\,\mathrm{h}).
\end{align*}
VerificaciУГn: $D(6.75) = 7 + 5\cos 0 = 12$ (marea alta) y
$D(12.75) = 7 + 5\cos\pi = 2$ (marea baja, medio perУ­odo despuУЉs), en
acuerdo con los datos.

\textbf{Paso 4 т ConclusiУГn y formalizaciУГn.}
\[
  \boxed{D(t) = 7.0 + 5.0\,\cos\!\left[\frac{\pi}{6}\,(t - 6.75)\right]
  \ \mathrm{m}.}
\]
\end{myproof}
\end{example}

\section{Funciones inversas}
% ============================================================

Muchos fenУГmenos fУ­sicos admiten dos perspectivas complementarias: dada la producciУГn de una fУЁbrica en funciУГn del tiempo, podrУ­amos querer saber en cuУЁnto tiempo se alcanza cierta producciУГn. Esto motiva el concepto de funciУГn inversa.

\begin{definition}[FunciУГn inyectiva]
$f: D_f \to \mathbb{R}$ es \textbf{inyectiva} (uno a uno) si
\[
  x_1 \neq x_2 \;\Rightarrow\; f(x_1) \neq f(x_2) \quad \forall\, x_1, x_2 \in D_f.
\]
Equivalentemente: $f(a) = f(b) \Rightarrow a = b$.
\end{definition}

\begin{rem}[Prueba de la recta horizontal]
$f$ es inyectiva si y solo si ninguna recta horizontal interseca su grУЁfica en mУЁs de un punto. Esta condiciУГn es necesaria y suficiente para que exista $f^{-1}$.
\end{rem}

\begin{definition}[FunciУГn inversa]
Sea $f: D_f \to R_f$ inyectiva. Su \textbf{funciУГn inversa} $f^{-1}: R_f \to D_f$ se define por
\[
  f^{-1}(y) = x \;\Longleftrightarrow\; f(x) = y,
\]
y satisface las identidades de composiciУГn:
\[
  f^{-1}(f(x)) = x \quad \forall\, x \in D_f, \qquad f(f^{-1}(y)) = y \quad \forall\, y \in R_f.
\]
\textbf{Advertencia:} $f^{-1} \neq 1/f$. La notaciУГn $f^{-1}$ denota la funciУГn inversa, \emph{no} el recУ­proco.
\end{definition}

\begin{rem}[Algoritmo para calcular $f^{-1}$]
Dado $y = f(x)$:
\begin{enumerate}
  \item Verificar que $f$ es inyectiva (prueba de la recta horizontal).
  \item Intercambiar los roles de $x$ e $y$: escribir $x = f(y)$.
  \item Despejar $y$ en tУЉrminos de $x$.
  \item El resultado es $y = f^{-1}(x)$.
\end{enumerate}
La grУЁfica de $f^{-1}$ es la reflexiУГn de la grУЁfica de $f$ respecto a la recta $y = x$ (figura~\ref{fig:inversa_reflexion}).
\end{rem}

\begin{example}
Determine la funciУГn inversa de
\[
  f(x) = \frac{4x-1}{2x+3} \qquad \text{y} \qquad g(x) = 2x - 3.
\]

\begin{myproof} Calculemos detalladamente la funciones inversas:
\textbf{Paso 1:} $f$ es racional de grado $1/1$; puede verificarse que es inyectiva en su dominio $\mathbb{R}\setminus\{-3/2\}$.

\textbf{Paso 2:} Escribimos $x = \dfrac{4y-1}{2y+3}$ y despejamos $y$:
\[
  x(2y+3) = 4y-1 \;\Rightarrow\; 2xy - 4y = -1-3x \;\Rightarrow\; y(2x-4)=-(1+3x).
\]

\textbf{Paso 3:} 
\[
  y = \frac{-(1+3x)}{2x-4} = \frac{3x+1}{4-2x}.
\]

\textbf{Paso 4:}
\[
  \boxed{f^{-1}(x) = \frac{3x+1}{4-2x}, \quad x \neq 2.}
\]

\medskip
\textbf{InversiУГn de $g$:}

\textbf{Pasos 1--3:} $x = 2y-3 \Rightarrow y = \dfrac{x+3}{2}$.

\textbf{Paso 4:}
\[
  \boxed{g^{-1}(x) = \frac{x+3}{2}.}

\end{myproof}
\end{example}

\begin{example} Si $f(x) = x^5 + x^3 + x$, determine $f^{-1}(3)$ y
$f(f^{-1}(2))$. AdemУЁs, si $g(x) = 3 + x + e^x$, determine $g^{-1}(4)$.

\begin{myproof}
\textbf{Paso 1 т AnУЁlisis y planteamiento.}
Despejar $y$ de $x = y^5 + y^3 + y$ es imposible con УЁlgebra elemental: no
hay fУГrmula explУ­cita para $f^{-1}$. Sin embargo, evaluar $f^{-1}$ en un
punto solo requiere responder: ТПquУЉ entrada produce esa salida?

\textbf{Paso 2 т FundamentaciУГn teУГrica.}
$f$ es suma de potencias impares, todas crecientes en $\mathbb{R}$, de modo
que $f$ es estrictamente creciente y, por tanto, inyectiva: $f^{-1}$ existe
y cada ecuaciУГn $f(x) = c$ tiene a lo sumo una soluciУГn. Lo mismo vale para
$g$, suma de la identidad (creciente), la exponencial (creciente) y una
constante. AdemУЁs, por la identidad de composiciУГn, $f(f^{-1}(y)) = y$
para todo $y \in R_f$.

\textbf{Paso 3 т Procedimiento tУЉcnico.}

\emph{CУЁlculo de $f^{-1}(3)$:} buscamos $x$ con $x^5+x^3+x = 3$. Ensayando
$x=1$: $1+1+1 = 3$. Por inyectividad es la УКnica soluciУГn, luego
$f^{-1}(3) = 1$.

\emph{CУЁlculo de $f(f^{-1}(2))$:} como $f$ es continua, creciente y no
acotada, $2 \in R_f$; la identidad de composiciУГn da directamente
$f(f^{-1}(2)) = 2$, sin necesidad de conocer el valor de $f^{-1}(2)$.

\emph{CУЁlculo de $g^{-1}(4)$:} buscamos $x$ con $3 + x + e^x = 4$.
Ensayando $x=0$: $3 + 0 + 1 = 4$. Por inyectividad, $g^{-1}(4) = 0$.

\textbf{Paso 4 т ConclusiУГn y formalizaciУГn.}
\[
  \boxed{f^{-1}(3) = 1, \qquad f(f^{-1}(2)) = 2, \qquad g^{-1}(4) = 0.}
\]
\end{myproof}
\end{example}

\begin{example} Sea $f(x) = \sqrt{1-x^2}$ con dominio restringido
$0 \leq x \leq 1$.
\begin{enumerate}
  \item Encuentre $f^{-1}$.
  \item ТПCУГmo se relaciona $f^{-1}$ con $f$?
  \item Identifique geomУЉtricamente la grУЁfica de $f$ y explique la
        relaciУГn hallada.
\end{enumerate}

\begin{myproof}
\textbf{Paso 1 т AnУЁlisis y planteamiento.}
Sin la restricciУГn, $f$ no serУ­a inyectiva (toma el mismo valor en $x$ y
$-x$). En $[0,1]$, $f$ decrece de $f(0)=1$ a $f(1)=0$, de modo que es
inyectiva, $R_f = [0,1]$ y la inversa existe.

\textbf{Paso 2 т FundamentaciУГn teУГrica.}
Se aplica el algoritmo de inversiУГn; al despejar la raУ­z se elige la rama
no negativa porque $f^{-1}: [0,1] \to [0,1]$ debe devolver valores del
dominio restringido.

\textbf{Paso 3 т Procedimiento tУЉcnico.}
De $x = \sqrt{1-y^2}$ con $0\le y\le 1$:
\[
  x^2 = 1 - y^2 \;\Rightarrow\; y^2 = 1 - x^2 \;\Rightarrow\;
  y = \sqrt{1-x^2}, \quad 0 \leq x \leq 1.
\]
Es decir, $f^{-1}$ tiene la misma fУГrmula y el mismo dominio que $f$: la
funciУГn es \emph{autoinversa}, $f^{-1} = f$.

\textbf{Paso 4 т ConclusiУГn y formalizaciУГn.}
La grУЁfica de $f$ es el cuarto de circunferencia unitaria del primer
cuadrante ($x^2+y^2=1$ con $x,y \geq 0$), que es simУЉtrico respecto a la
recta $y=x$: al reflejarla queda idУЉntica, y por eso $f^{-1}=f$ (el
mecanismo general de reflexiУГn se ilustra en la
figura~\ref{fig:inversa_reflexion}).
\[
  \boxed{f^{-1}(x) = \sqrt{1-x^2},\; 0\le x\le 1; \qquad f^{-1} = f.}
\]
\end{myproof}
\end{example}

\section{Operaciones con exponenciales y logaritmos}
% ============================================================

Las leyes que gobiernan las operaciones con potencias y logaritmos son la base del УЁlgebra del cУЁlculo. Las resumimos en los siguientes teoremas.

\begin{theorem}[Leyes de los exponentes]
Sean $a, b > 0$ y $x, y \in \mathbb{R}$. Entonces:
\begin{enumerate}
  \item $b^{x+y} = b^x \cdot b^y$
  \item $b^{x-y} = b^x / b^y$
  \item $(b^x)^y = b^{xy}$
  \item $(ab)^x = a^x b^x$
\end{enumerate}
\end{theorem}

\begin{theorem}[Leyes de los logaritmos]
Sea $b > 0$, $b \neq 1$, y $x, y > 0$, $r \in \mathbb{R}$. Entonces:
\begin{enumerate}
  \item $\log_b(xy) = \log_b x + \log_b y$
  \item $\log_b(x/y) = \log_b x - \log_b y$
  \item $\log_b(x^r) = r\log_b x$
\end{enumerate}
AdemУЁs: $\log_b x = y \Leftrightarrow b^y = x$.
\end{theorem}

\begin{definition}[Logaritmo natural]
$\ln x = \log_e x$ donde $e \approx 2.71828$. Satisface $\ln(e^x) = x$ y $e^{\ln x} = x$ para $x > 0$. En particular, $\ln e = 1$ y $\ln 1 = 0$.
\end{definition}

\begin{theorem}[FУГrmula de cambio de base]
Sea $b > 0$, $b \neq 1$. Entonces
\[
  \log_b x = \frac{\ln x}{\ln b}.
\]
\end{theorem}
\begin{proof}
Sea $y = \log_b x$, es decir $b^y = x$. Aplicando $\ln$ a ambos lados: $y \ln b = \ln x$. Despejando: $y = \ln x / \ln b$. \qed
\end{proof}
