\chapter{Funciones}\label{funciones}

La función es, sin duda, el concepto más omnipresente en las matemáticas: toda la arquitectura del cálculo diferencial e integral descansa sobre él. Antes de precisar la definición formal, conviene cultivar la intuición correcta. Imagine una \emph{máquina procesadora de números}: se introduce un valor $x$ en la ranura de entrada, la máquina ejecuta una operación interna y devuelve un único resultado $f(x)$ por la ranura de salida. Esta metáfora captura la esencia de la dependencia funcional: el valor de salida \emph{depende} del valor de entrada, y a cada entrada le corresponde exactamente una salida.

En la práctica, esta dependencia aparece en todas partes: el área de un círculo depende de su radio, el costo de un envío depende de su peso, la temperatura de un cuerpo depende del tiempo transcurrido. En este capítulo desarrollaremos con rigor y detalle este concepto fundamental, explorando sus propiedades algebraicas y su comportamiento gráfico.


\section{Propiedades básicas}

\begin{definition}[Función]
Una \textbf{función} $f$ es una regla de correspondencia que asigna a cada elemento $x$ de un conjunto denominado \textbf{dominio} $D_f$, un \emph{único} valor $f(x)$ en otro conjunto llamado \textbf{codominio}. 

El conjunto de todos los valores efectivamente alcanzados,
\[
  R_f = \{\, f(x) : x \in D_f \,\},
\]
se denomina \textbf{rango} o \textbf{imagen} de $f$, y satisface $R_f \subseteq \text{codominio}$. Escribimos $f: D_f \to \mathbb{R}$ para indicar que $f$ va del dominio al conjunto de los números reales. Es convención llamar a $x$ la \textbf{variable independiente} y a $y = f(x)$ la \textbf{variable dependiente}.
\end{definition}

\begin{rem}[La regla del único valor de salida]
La unicidad del valor de salida es la condición definitoria de una función. Si a un valor $x$ le correspondieran dos salidas distintas $y_1 \neq y_2$, la regla no sería una función. Esta unicidad se verifica gráficamente mediante la \emph{prueba de la recta vertical}.
\end{rem}

\subsection{Dominio natural de una función}
% -----------------------------------------------------------

Cuando una función se presenta mediante una fórmula algebraica sin restricciones explícitas sobre su dominio, el \textbf{dominio natural} (o dominio implícito) es el conjunto más amplio de números reales para los cuales la expresión está definida. Las dos fuentes principales de restricción son:

\begin{itemize}
  \item \textbf{Denominadores:} se excluyen los valores que anulan el denominador.
  \item \textbf{Radicales de índice par:} se requiere que el radicando sea no negativo.
\end{itemize}



\begin{prob}\label{prob:dominio-basico}
Calcule el dominio natural de cada función y expréselo en notación de intervalos.
\begin{enumerate}
  \item $f(x) = x^2 - 8x + 5$
  \item $f(x) = \sqrt{1 - x^2}$
  \item $f(x) = \sqrt{3+x} + \sqrt[4]{7-x}$
  \item $f(x) = \dfrac{x+5}{x-3}$
\end{enumerate}

\begin{myproof}
\textbf{1.} \emph{Análisis:} los polinomios no tienen restricciones.

\emph{Fundamentación:} la función es un polinomio de grado $2$; los polinomios están definidos para todo $x \in \mathbb{R}$.

\emph{Conclusión:}
\[
  \boxed{D_f = \mathbb{R} = (-\infty,+\infty).}
\]

\medskip
\textbf{2.} \emph{Análisis:} la raíz cuadrada requiere radicando no negativo, lo que acotará $x$ a un intervalo finito.

\emph{Fundamentación:} necesitamos $1 - x^2 \geq 0$. Factorizando:
\[
  1 - x^2 = (1-x)(1+x) \geq 0.
\]

\emph{Procedimiento:} construimos la tabla de signos:

\begin{table}[H]
\centering
\begin{tabular}{|c|c|c|c|}
\hline
Intervalo & $1+x$ & $1-x$ & $(1-x)(1+x)$ \\
\hline
$(-\infty,-1)$ & $-$ & $+$ & $-$ \\
\hline
$[-1,\,1]$      & $+$ & $+$ & $+$ \\
\hline
$(1,+\infty)$   & $+$ & $-$ & $-$ \\
\hline
\end{tabular}
\caption{Tabla de signos para el ítem 2 del Problema \ref{prob:dominio-basico}.}
\end{table}

\emph{Conclusión:}
\[
  \boxed{D_f = [-1,\,1].}
\]

\medskip
\textbf{3.} \emph{Análisis:} dos radicales de índice par imponen dos condiciones simultáneas.

\emph{Procedimiento:} la raíz cuarta tiene índice par, por lo que también requiere radicando no negativo.
\begin{align*}
  \sqrt{3+x}:   &\quad 3 + x \geq 0 \;\Rightarrow\; x \geq -3, \\
  \sqrt[4]{7-x}:&\quad 7 - x \geq 0 \;\Rightarrow\; x \leq 7.
\end{align*}
La suma está definida cuando \emph{ambas} condiciones se cumplen.

\emph{Conclusión:}
\[
  \boxed{D_f = [-3,\,7].}
\]

\medskip
\textbf{4.} \emph{Análisis:} el denominador no puede ser cero.

\emph{Procedimiento:} el denominador se anula en $x = 3$.

\emph{Conclusión:}
\[
  \boxed{D_f = \mathbb{R} \setminus \{3\} = (-\infty, 3)\cup(3,+\infty).}
\]
\end{myproof}
\end{prob}

\subsection{Cuatro formas de representar una función}
% -----------------------------------------------------------

\begin{rem}[Formas de representar una función]
Existen cuatro formas complementarias de representar una función:
\begin{enumerate}
  \item \textbf{Verbal:} descripción en palabras de la relación entre variables.
  \item \textbf{Numérica:} tabla de valores $(x, f(x))$.
  \item \textbf{Algebraica:} fórmula explícita $y = f(x)$.
  \item \textbf{Visual:} gráfica en el plano $xy$.
\end{enumerate}
Cada representación revela aspectos distintos del comportamiento de la función; la habilidad del matemático consiste en elegir la más conveniente para cada situación.
\end{rem}


\begin{example}\label{ex:costo-contenedor}
Un contenedor rectangular sin tapa tiene un volumen de $10\,\mathrm{m}^3$. La longitud de su base es el doble de su ancho. El material para la base cuesta \$10 por metro cuadrado y el material para los cuatro lados cuesta \$6 por metro cuadrado. Exprese el costo de los materiales como función del ancho de la base, proporcionando las cuatro representaciones.

\begin{myproof}
\textbf{Paso 1 — Análisis gráfico y tabular.} Sea $x > 0$ el ancho de la base (en metros). Entonces la longitud es $2x$ y la altura $h$ se determinará por la condición de volumen. Anticipamos que $C(x)$ crecerá cuando $x \to 0^+$ (los lados se vuelven muy altos) y cuando $x \to +\infty$ (la base domina el costo), por lo que tendrá un mínimo.

\textbf{Paso 2 — Fundamentación.} Las áreas de las superficies determinan el costo; la condición de volumen fija $h$ en función de $x$.

\textbf{Paso 3 — Procedimiento algebraico.} De la condición de volumen:
\[
  V = x \cdot 2x \cdot h = 2x^2 h = 10 \;\Rightarrow\; h = \frac{5}{x^2}.
\]
Áreas y costos parciales:
\begin{align*}
  A_{\text{base}} &= 2x^2, & C_{\text{base}} &= 10 \cdot 2x^2 = 20x^2,\\
  A_{\text{lados}} &= 2(xh) + 2(2xh) = 6xh = \frac{30}{x}, & C_{\text{lados}} &= 6 \cdot \frac{30}{x} = \frac{180}{x}.
\end{align*}

\textbf{Paso 4 — Conclusión y representaciones.}

\textbf{Algebraica:}
\[
  \boxed{C(x) = 20x^2 + \frac{180}{x},\quad x > 0.}
\]

\textbf{Verbal:} A cada ancho positivo $x$ de la base se asocia el costo total de construir el contenedor, sumando el aporte de la base cuadrada (que crece con $x^2$) y el de los cuatro lados (que decrecen con $x$).

\textbf{Numérica:}
\[
\begin{array}{c|c}
x\,(\mathrm{m}) & C(x)\,(\$) \\\hline
1   & 200 \\
1.5 & 165 \\
2   & 170 \\
2.5 & 197 \\
\end{array}
\]

\textbf{Visual:}
\begin{figure}[H]
\centering
\begin{tikzpicture}[scale=0.85]
\begin{axis}[
    axis lines = left,
    xlabel={$x$ (ancho en metros)},
    ylabel={$C(x)$ (costo en \$)},
    xmin=0.5, xmax=4,
    ymin=100, ymax=310,
    grid=major,
    xtick={1,1.5,2,2.5,3,3.5},
    ytick={150,200,250,300},
    domain=0.55:3.9,
    samples=120,
    width=11cm, height=7cm,
    legend pos=north east,
]
\addplot[blue, ultra thick] {20*x^2 + 180/x};
\addlegendentry{$C(x)=20x^2+\frac{180}{x}$}
\addplot[red, mark=*, only marks, mark size=2.5pt] coordinates {
    (1,200)(1.5,165)(2,170)(2.5,197)};
\addlegendentry{Puntos de la tabla}
\node[pin=80:{$(1.5,165)$}] at (axis cs:1.5,165) {};
\end{axis}
\end{tikzpicture}
\caption{Gráfica de $C(x)$ del Ejemplo \ref{ex:costo-contenedor}: el mínimo ocurre aproximadamente en $x \approx 1.65\,\mathrm{m}$.}
\end{figure}
\end{myproof}
\end{example}

Aunque conocer las cuatro representaciones de una función (verbal, numérica, algebraica y gráfica) es fundamental, en ocasiones una resulta más conveniente que las demás según el contexto. Por ello, es esencial seleccionar la representación más adecuada para cada situación particular. La siguiente es una propiedad clave para identificar gráficas de funciones:

\begin{rem}[Prueba de la recta vertical]
Una curva en el plano $xy$ es la gráfica de una función $y = f(x)$ \emph{si y solo si} ninguna recta vertical interseca la curva en más de un punto. Por ejemplo, la circunferencia $x^2 + y^2 = 1$ no es la gráfica de una función: la recta $x = 0.5$ la corta en dos puntos, $(0.5,\,\sqrt{3}/2)$ y $(0.5,\,-\sqrt{3}/2)$.

\begin{figure}[H]
\centering
\begin{tikzpicture}[scale=1.5]
\draw[->] (-1.5,0) -- (1.5,0) node[right] {$x$};
\draw[->] (0,-1.5) -- (0,1.5) node[above] {$y$};
\draw[thick] (0,0) circle (1cm);
\draw[red, very thick] (0.5,-1.4) -- (0.5,1.4) node[above right, font=\small] {$x=0.5$};
\filldraw[red] (0.5,{sqrt(3)/2}) circle (2pt) node[right, font=\small] {$(0.5,\frac{\sqrt{3}}{2})$};
\filldraw[red] (0.5,{-sqrt(3)/2}) circle (2pt) node[right, font=\small] {$(0.5,-\frac{\sqrt{3}}{2})$};
\end{tikzpicture}
\caption{La circunferencia unitaria falla la prueba de la recta vertical.}
\end{figure}
\end{rem}

\subsection{Funciones crecientes, decrecientes, pares e impares}
% -----------------------------------------------------------

\begin{definition}[Funciones monótonas]
Sea $f$ una función definida en un intervalo $I$.
\begin{itemize}
  \item $f$ es \textbf{creciente} en $I$ si $x_1 < x_2 \Rightarrow f(x_1) < f(x_2)$ para todo $x_1, x_2 \in I$.
  \item $f$ es \textbf{decreciente} en $I$ si $x_1 < x_2 \Rightarrow f(x_1) > f(x_2)$ para todo $x_1, x_2 \in I$.
\end{itemize}
\end{definition}

\begin{example}
Determine los intervalos de crecimiento y decrecimiento de $f(x) = x^2$.

\begin{myproof}
\textbf{Paso 1 — Análisis gráfico.} La parábola tiene vértice en el origen y es simétrica respecto al eje $y$; se anticipa crecimiento en $[0,+\infty)$ y decrecimiento en $(-\infty,0]$.

\textbf{Paso 2 — Fundamentación.} Usaremos la definición: analizamos el signo de $f(x_2) - f(x_1)$ para $x_1 < x_2$.

\textbf{Paso 3 — Procedimiento.}
\[
  f(x_2) - f(x_1) = x_2^2 - x_1^2 = (x_2 - x_1)(x_2 + x_1).
\]
Como $x_1 < x_2$, se tiene $x_2 - x_1 > 0$. El signo de la diferencia depende de $x_2 + x_1$:
\begin{itemize}
  \item Si $x_1, x_2 \geq 0$: $x_2 + x_1 \geq 0$, luego $f(x_2) - f(x_1) \geq 0$. ($f$ es creciente.)
  \item Si $x_1, x_2 \leq 0$: $x_2 + x_1 \leq 0$, luego $f(x_2) - f(x_1) \leq 0$. ($f$ es decreciente.)
\end{itemize}

\textbf{Paso 4 — Conclusión:}
\[
  \boxed{f \text{ es decreciente en } (-\infty, 0] \text{ y creciente en } [0,+\infty).}
\]
\end{myproof}
\end{example}

El ejemplo anterior se resolverá con mayor facilidad más adelante, luego de estudiar la derivada. Mientras tanto, observe que hay algunas formas adicionales de representar funciones:

\begin{definition}[Funciones pares e impares]
Sea $f: D_f \to \mathbb{R}$ con $D_f$ simétrico respecto al origen (es decir, $x \in D_f \Rightarrow -x \in D_f$).
\begin{itemize}
  \item $f$ es \textbf{par} si $f(-x) = f(x)$ para todo $x \in D_f$. Su gráfica es simétrica respecto al eje $y$.
  \item $f$ es \textbf{impar} si $f(-x) = -f(x)$ para todo $x \in D_f$. Su gráfica es simétrica respecto al origen.
\end{itemize}
\end{definition}

\begin{example}
Clasifique $f(x) = x^2$ y $g(x) = x^3$ según paridad.

\begin{myproof}
Para $f$: $f(-x) = (-x)^2 = x^2 = f(x)$, luego $f$ es \textbf{par}. \quad
Para $g$: $g(-x) = (-x)^3 = -x^3 = -g(x)$, luego $g$ es \textbf{impar}. \qed
\end{myproof}
\end{example}

\subsection{Función definida por partes}
% -----------------------------------------------------------

\begin{example}[Función por tramos]
Trace la gráfica de
\[
  f(x) = \begin{cases} 3 - \dfrac{1}{2}x & \text{si } x < 2, \\ 2x - 5 & \text{si } x \geq 2. \end{cases}
\]

\begin{myproof}
\textbf{Paso 1 — Análisis:} dos rectas que se empalman en $x=2$; verificamos continuidad:
\[
  \lim_{x\to 2^-} f(x) = 3 - 1 = 2, \qquad f(2) = 2(2)-5 = -1.
\]
Hay un salto en $x = 2$.

\textbf{Paso 2:} Para $x < 2$: recta decreciente (pendiente $-\tfrac{1}{2}$), \emph{círculo abierto} en $(2, 2)$. Para $x \geq 2$: recta creciente (pendiente $2$), \emph{punto sólido} en $(2,-1)$.

\begin{figure}[H]
\centering
\begin{tikzpicture}[scale=1.2]
\draw[->] (-0.5,0) -- (4.5,0) node[right] {$x$};
\draw[->] (0,-2) -- (0,4) node[above] {$y$};
\draw[thick, blue, domain=-0.5:1.99, samples=80] plot (\x,{3-0.5*\x});
\fill[white] (2,2) circle (3pt); \draw[blue,thick] (2,2) circle (3pt);
\draw[thick, red, domain=2:4.2, samples=60] plot (\x,{2*\x-5});
\filldraw[red] (2,-1) circle (3pt);
\node[blue, above left, font=\small] at (0.5,2.8) {$3-\tfrac{1}{2}x\ (x<2)$};
\node[red, right, font=\small] at (3.5,2.2) {$2x-5\ (x\geq2)$};
\draw[dashed, gray] (2,-2) -- (2,3.5);
\node[below, font=\small] at (2,0) {$2$};
\end{tikzpicture}
\caption{Función por tramos con salto en $x=2$.}
\end{figure}

\textbf{Conclusión:} $D_f = \mathbb{R}$; la función tiene un salto en $x = 2$ de magnitud $|{-1} - 2| = 3$. \qed
\end{myproof}
\end{example}

\section{Catálogo de funciones elementales fundamentales}
% ============================================================

Existe un puñado de funciones básicas ---las herramientas fundamentales del cálculo--- a partir de las cuales se construyen virtualmente todos los modelos matemáticos. Las estudiaremos sistemáticamente.

% -----------------------------------------------------------
\subsection{Función potencia}
% -----------------------------------------------------------

\begin{definition}[Función potencia]
Sea $r \in \mathbb{Q}$, $r = p/q$ en forma reducida con $q \in \mathbb{N}^+$. La \textbf{función potencia} es $f(x) = x^r = \sqrt[q]{x^p}$. Su dominio depende de la paridad de $q$ y el signo de $r$:
\begin{itemize}
  \item $q$ \textbf{impar}: $D_f = \mathbb{R}$.
  \item $q$ \textbf{par}: requiere $x^p \geq 0$; para $p$ impar, $D_f = [0,+\infty)$.
  \item $r < 0$: se excluye $x = 0$: $D_f = \mathbb{R}\setminus\{0\}$ (o $(0,+\infty)$ si $q$ par).
\end{itemize}
\end{definition}

\begin{rem}
Como casos particulares, para enteros positivos $n \in \mathbb{N}$, $f(x) = x^n$ es un polinomio definido en todo $\mathbb{R}$. Para enteros negativos $n < 0$, $f(x) = x^n = \frac{1}{x^{-n}}$ está definida en $\mathbb{R} \setminus \{0\}$.

Estas reglas garantizan que $x^r$ produzca valores reales para todo $x$ en su dominio respectivo.
\end{rem}

\subsection{Funciones exponenciales y logarítmicas}
% -----------------------------------------------------------

\begin{definition}[Función exponencial]
Sea $a > 0$, $a \neq 1$. La \textbf{función exponencial} de base $a$ es $f(x) = a^x$, con $D_f = \mathbb{R}$ y $R_f = (0,+\infty)$. Es estrictamente creciente si $a > 1$ y estrictamente decreciente si $0 < a < 1$. Tiene asíntota horizontal $y = 0$ y pasa por $(0,1)$.

El número $e \approx 2.71828\ldots$ (base de Euler) ocupa un lugar privilegiado: la función $e^x$ es la única cuya pendiente en $(0,1)$ es exactamente $1$, propiedad que se demostrará formalmente al estudiar la derivada.
\end{definition}

\begin{definition}[Función logarítmica]
Sea $a > 0$, $a \neq 1$. La \textbf{función logarítmica} de base $a$ es $f(x) = \log_a x$, definida como la inversa de $a^x$:
\[
  a^{\log_a x} = x \quad (x > 0) \qquad \text{y} \qquad \log_a(a^x) = x \quad (x \in \mathbb{R}).
\]
$D_f = (0,+\infty)$, $R_f = \mathbb{R}$. Pasa por $(1,0)$ y tiene asíntota vertical $x = 0$. Es creciente si $a > 1$ y decreciente si $0 < a < 1$.
\end{definition}

\subsection{Funciones trigonométricas}
% -----------------------------------------------------------

\begin{definition}[Función periódica]
$f: D_f \to \mathbb{R}$ es \textbf{periódica} si existe $T > 0$ tal que $f(x+T) = f(x)$ para todo $x \in D_f$ con $x+T \in D_f$. El menor $T$ se llama \textbf{período fundamental}.
\end{definition}

\begin{definition}[Funciones trigonométricas fundamentales]
Las seis funciones trigonométricas fundamentales se definen con $x \in \mathbb{R}$ en radianes:

\begin{center}

\subsection{Función compuesta y función elemental}
% -----------------------------------------------------------

\begin{definition}[Función compuesta]
Sean $f: D_f \to \mathbb{R}$ y $g: D_g \to \mathbb{R}$. La \textbf{composición} $f \circ g$ se define por
\[
  (f \circ g)(x) = f(g(x)),
\]
con dominio $D_{f \circ g} = \{\, x \in D_g : g(x) \in D_f \,\}$.
\end{definition}

\begin{example}
Identifique la composición en $y = \sqrt{1-x^2}$ y halle su dominio.

\begin{myproof}
\textbf{Paso 1:} La función exterior es $f(u) = \sqrt{u}$ (requiere $u \geq 0$); la interior es $g(x) = 1-x^2$.

\textbf{Paso 2:} Para $f\circ g$ se necesita $g(x) \in D_f = [0,+\infty)$.

\textbf{Paso 3:} $1-x^2 \geq 0 \Leftrightarrow x^2 \leq 1 \Leftrightarrow -1 \leq x \leq 1$.

\textbf{Paso 4:}
\[
  \boxed{D_{f\circ g} = [-1,\,1].}
\]
\end{myproof}
\end{example}

\begin{rem}[Composiciones múltiples]
La función $h(x) = \ln(\sin(x^2+1))$ es la composición $h = \ln \circ \sin \circ p$, donde $p(x)=x^2+1$. Para hallar su dominio se imponen las restricciones de adentro hacia afuera: $p(x)=x^2+1 \geq 1 > 0$ (sin restricción); $\sin(p(x))$ debe ser positivo para que $\ln$ esté definido. Luego
\[
  D_h = \{\, x \in \mathbb{R} : \sin(x^2+1) > 0 \,\}.
\]
\end{rem}

\begin{definition}[Función elemental]
Una función $f$ es \textbf{elemental} si se obtiene a partir de funciones potencia, exponenciales, logarítmicas, trigonométricas (y sus inversas) y constantes reales mediante un número \emph{finito} de sumas, restas, multiplicaciones, divisiones y composiciones.

Las \textbf{funciones algebraicas} son el subconjunto de funciones elementales que solo involucran operaciones algebraicas (sin exponenciales ni trigonométricas). Se subdividen en:
\begin{itemize}
  \item \textbf{Polinomios:} $f(x) = a_n x^n + \cdots + a_0$, $D_f = \mathbb{R}$.
  \item \textbf{Racionales:} $f(x) = P(x)/Q(x)$, $D_f = \{x : Q(x)\neq 0\}$.
  \item \textbf{Irracionales:} contienen raíces $\sqrt[n]{\cdot}$ con $n \geq 2$.
\end{itemize}
Las funciones no algebraicas (senos, cosenos, exponenciales, logaritmos) se denominan \textbf{trascendentes}.
\end{definition}

\section{Transformaciones de funciones}
% ============================================================

A partir de una función base $y = f(x)$, podemos generar una familia entera de funciones mediante \textbf{transformaciones geométricas} que modifican la posición y escala de la gráfica sin alterar su forma esencial.

% -----------------------------------------------------------
\subsection{Traslaciones}
% -----------------------------------------------------------

Dados $h, k \in \mathbb{R}$:
\begin{itemize}
  \item $y = f(x-h)$: \textbf{traslación horizontal} $h$ unidades ($h>0$: derecha; $h<0$: izquierda).
  \item $y = f(x)+k$: \textbf{traslación vertical} $k$ unidades ($k>0$: arriba; $k<0$: abajo).
  \item $y = f(x-h)+k$: \textbf{traslación combinada}.
\end{itemize}

\subsection{Dilataciones, compresiones y reflexiones}
% -----------------------------------------------------------

Sea $c > 0$, $c \neq 1$:
\begin{itemize}
  \item $y = c\,f(x)$, $c>1$: \textbf{dilatación vertical} (estira hacia arriba).
  \item $y = \tfrac{1}{c}\,f(x)$, $c>1$: \textbf{compresión vertical}.
  \item $y = f(cx)$, $c>1$: \textbf{compresión horizontal} (acorta el período).
  \item $y = f(x/c)$, $c>1$: \textbf{dilatación horizontal} (alarga el período).
  \item $y = -f(x)$: \textbf{reflexión respecto al eje $x$}.
  \item $y = f(-x)$: \textbf{reflexión respecto al eje $y$} (no altera paridad).
\end{itemize}

Para $\sin$ y $\cos$, la forma generalizada
\[
  y = C + A\sin\bigl[a(t - b)\bigr] \quad \text{o} \quad y = C + A\cos\bigl[a(t - b)\bigr]
\]
tiene los siguientes parámetros geométricos:
\begin{itemize}
  \item \textbf{Amplitud:} $|A|$ (distancia desde la línea central hasta los extremos).
  \item \textbf{Período fundamental:} $T = 2\pi/|a|$.
  \item \textbf{Línea central (nivel medio):} $y = C$.
  \item \textbf{Desfase horizontal:} $b$ (desplaza el ciclo).
\end{itemize}

\section{Funciones inversas}
% ============================================================

Muchos fenómenos físicos admiten dos perspectivas complementarias: dada la producción de una fábrica en función del tiempo, podríamos querer saber en cuánto tiempo se alcanza cierta producción. Esto motiva el concepto de función inversa.

\begin{definition}[Función inyectiva]
$f: D_f \to \mathbb{R}$ es \textbf{inyectiva} (uno a uno) si
\[
  x_1 \neq x_2 \;\Rightarrow\; f(x_1) \neq f(x_2) \quad \forall\, x_1, x_2 \in D_f.
\]
Equivalentemente: $f(a) = f(b) \Rightarrow a = b$.
\end{definition}

\begin{rem}[Prueba de la recta horizontal]
$f$ es inyectiva si y solo si ninguna recta horizontal interseca su gráfica en más de un punto. Esta condición es necesaria y suficiente para que exista $f^{-1}$.
\end{rem}

\begin{definition}[Función inversa]
Sea $f: D_f \to R_f$ inyectiva. Su \textbf{función inversa} $f^{-1}: R_f \to D_f$ se define por
\[
  f^{-1}(y) = x \;\Longleftrightarrow\; f(x) = y,
\]
y satisface las identidades de composición:
\[
  f^{-1}(f(x)) = x \quad \forall\, x \in D_f, \qquad f(f^{-1}(y)) = y \quad \forall\, y \in R_f.
\]
\textbf{Advertencia:} $f^{-1} \neq 1/f$. La notación $f^{-1}$ denota la función inversa, \emph{no} el recíproco.
\end{definition}

\begin{rem}[Algoritmo para calcular $f^{-1}$]
Dado $y = f(x)$:
\begin{enumerate}
  \item Verificar que $f$ es inyectiva (prueba de la recta horizontal).
  \item Intercambiar los roles de $x$ e $y$: escribir $x = f(y)$.
  \item Despejar $y$ en términos de $x$.
  \item El resultado es $y = f^{-1}(x)$.
\end{enumerate}
La gráfica de $f^{-1}$ es la reflexión de la gráfica de $f$ respecto a la recta $y = x$.
\end{rem}

\begin{example}
Determine la función inversa de
\[
  f(x) = \frac{4x-1}{2x+3} \qquad \text{y} \qquad g(x) = 2x - 3.
\]

\begin{myproof} Calculemos detalladamente la funciones inversas:
\textbf{Paso 1:} $f$ es racional de grado $1/1$; puede verificarse que es inyectiva en su dominio $\mathbb{R}\setminus\{-3/2\}$.

\textbf{Paso 2:} Escribimos $x = \dfrac{4y-1}{2y+3}$ y despejamos $y$:
\[
  x(2y+3) = 4y-1 \;\Rightarrow\; 2xy - 4y = -1-3x \;\Rightarrow\; y(2x-4)=-(1+3x).
\]

\textbf{Paso 3:} 
\[
  y = \frac{-(1+3x)}{2x-4} = \frac{3x+1}{4-2x}.
\]

\textbf{Paso 4:}
\[
  \boxed{f^{-1}(x) = \frac{3x+1}{4-2x}, \quad x \neq 2.}
\]

\medskip
\textbf{Inversión de $g$:}

\textbf{Pasos 1--3:} $x = 2y-3 \Rightarrow y = \dfrac{x+3}{2}$.

\textbf{Paso 4:}
\[
  \boxed{g^{-1}(x) = \frac{x+3}{2}.}
\]
\end{myproof}
\end{example}

\section{Operaciones con exponenciales y logaritmos}
% ============================================================

Las leyes que gobiernan las operaciones con potencias y logaritmos son la base del álgebra del cálculo. Las resumimos en los siguientes teoremas.

\begin{theorem}[Leyes de los exponentes]
Sean $a, b > 0$ y $x, y \in \mathbb{R}$. Entonces:
\begin{enumerate}
  \item $b^{x+y} = b^x \cdot b^y$
  \item $b^{x-y} = b^x / b^y$
  \item $(b^x)^y = b^{xy}$
  \item $(ab)^x = a^x b^x$
\end{enumerate}
\end{theorem}

\begin{theorem}[Leyes de los logaritmos]
Sea $b > 0$, $b \neq 1$, y $x, y > 0$, $r \in \mathbb{R}$. Entonces:
\begin{enumerate}
  \item $\log_b(xy) = \log_b x + \log_b y$
  \item $\log_b(x/y) = \log_b x - \log_b y$
  \item $\log_b(x^r) = r\log_b x$
\end{enumerate}
Además: $\log_b x = y \Leftrightarrow b^y = x$.
\end{theorem}

\begin{definition}[Logaritmo natural]
$\ln x = \log_e x$ donde $e \approx 2.71828$. Satisface $\ln(e^x) = x$ y $e^{\ln x} = x$ para $x > 0$. En particular, $\ln e = 1$ y $\ln 1 = 0$.
\end{definition}

\begin{theorem}[Fórmula de cambio de base]
Sea $b > 0$, $b \neq 1$. Entonces
\[
  \log_b x = \frac{\ln x}{\ln b}.
\]
\end{theorem}
\begin{proof}
Sea $y = \log_b x$, es decir $b^y = x$. Aplicando $\ln$ a ambos lados: $y \ln b = \ln x$. Despejando: $y = \ln x / \ln b$. \qed
\end{proof}
